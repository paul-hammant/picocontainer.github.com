011package com.picocontainer.parameters;
012
013import java.io.Serializable;
014import java.lang.annotation.Annotation;
015import java.lang.reflect.Field;
016import java.lang.reflect.ParameterizedType;
017import java.lang.reflect.Type;
018
019import com.picocontainer.ComponentAdapter;
020import com.picocontainer.NameBinding;
021import com.picocontainer.Parameter;
022import com.picocontainer.PicoCompositionException;
023import com.picocontainer.PicoContainer;
024import com.picocontainer.PicoException;
025import com.picocontainer.PicoVisitor;
026
027
028/**
029 * A ConstantParameter should be used to pass in "constant" arguments to constructors. This
030 * includes {@link String}s,{@link Integer}s or any other object that is not registered in
031 * the container.
032 *
033 * @author Jon Tirs&eacute;n
034 * @author Aslak Helles&oslash;y
035 * @author J&ouml;rg Schaible
036 * @author Thomas Heller
037 */
038@SuppressWarnings("serial")
039public class ConstantParameter extends AbstractParameter implements Parameter, Serializable {
040
041    private final Object value;
042
043
044    public ConstantParameter(final Object value) {
045                this.value = value;
046
047    }
048
049    public Resolver resolve(final PicoContainer container, final ComponentAdapter<?> forAdapter,
050                            final ComponentAdapter<?> injecteeAdapter, final Type expectedType, final NameBinding expectedNameBinding,
051                            final boolean useNames, final Annotation binding) {
052        if (expectedType instanceof Class) {
053            return new Parameter.ValueResolver(isAssignable(expectedType), value, null);
054        } else if (expectedType instanceof ParameterizedType) {
055                return new Parameter.ValueResolver(isAssignable(((ParameterizedType)expectedType).getRawType()), value, null);
056        }
057        return new Parameter.ValueResolver(true, value, null);
058    }
059
060    /**
061     * {@inheritDoc}
062     *
063     * @see Parameter#verify(com.picocontainer.PicoContainer,com.picocontainer.ComponentAdapter,java.lang.reflect.Type,com.picocontainer.NameBinding,boolean,java.lang.annotation.Annotation)
064     */
065    public void verify(final PicoContainer container, final ComponentAdapter<?> adapter,
066                       final Type expectedType, final NameBinding expectedNameBinding,
067                       final boolean useNames, final Annotation binding) throws PicoException {
068        if (!isAssignable(expectedType)) {
069            throw new PicoCompositionException(
070                expectedType + " is not assignable from " +
071                        (value != null ? value.getClass().getName() : "null"));
072        }
073    }
074
075    protected boolean isAssignable(final Type expectedType) {
076        if (expectedType instanceof Class) {
077            Class<?> expectedClass = (Class<?>) expectedType;
078            if (checkPrimitive(expectedClass) || expectedClass.isInstance(value)) {
079                return true;
080            }
081        }
082        return false;
083    }
084
085    /**
086     * Visit the current {@link Parameter}.
087     *
088     * @see com.picocontainer.Parameter#accept(com.picocontainer.PicoVisitor)
089     */
090    public void accept(final PicoVisitor visitor) {
091        visitor.visitParameter(this);
092    }
093
094    private boolean checkPrimitive(final Class<?> expectedType) {
095        try {
096            if (expectedType.isPrimitive()) {
097                final Field field = value.getClass().getField("TYPE");
098                final Class<?> type = (Class<?>) field.get(value);
099                return expectedType.isAssignableFrom(type);
100            }
101        } catch (NoSuchFieldException e) {
102            //ignore
103        } catch (IllegalAccessException e) {
104            //ignore
105        }
106        return false;
107    }
108
109
110        public Object getValue() {
111                return value;
112        }
113
114}
