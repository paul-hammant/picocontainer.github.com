041/**
042 * A CollectionComponentParameter should be used to support inject an
043 * {@link Array}, a {@link Collection}or {@link Map}of components automatically.
044 * The collection will contain all components of a special type and additionally
045 * the type of the key may be specified. In case of a map, the map's keys are
046 * the one of the component adapter.
047 *
048 * @author Aslak Helles&oslash;y
049 * @author J&ouml;rg Schaible
050 */
051@SuppressWarnings("serial")
052public class CollectionComponentParameter extends AbstractParameter implements Parameter, Serializable {
053
054        /**
055         * Use <code>ARRAY</code> as {@link Parameter}for an Array that must have
056         * elements.
057         */
058        public static final CollectionComponentParameter ARRAY = new CollectionComponentParameter();
059        /**
060         * Use <code>ARRAY_ALLOW_EMPTY</code> as {@link Parameter}for an Array that
061         * may have no elements.
062         */
063        public static final CollectionComponentParameter ARRAY_ALLOW_EMPTY = new CollectionComponentParameter(true);

065        private final boolean emptyCollection;
066        private final Class<?> keyType;
067        private final Generic<?> componentValueType;

119        /**
120         * Check for a successful dependency resolution of the parameter for the
121         * expected type. The dependency can only be satisfied if the expected type
122         * is one of the collection types {@link Array},{@link Collection}or
123         * {@link Map}. An empty collection is only a valid resolution, if the
124         * <code>emptyCollection</code> flag was set.
125         *
126         * @param container
127         *            {@inheritDoc}
128         * @param injecteeAdapter
129         * @param expectedType
130         *            {@inheritDoc}
131         * @param expectedNameBinding
132         *            {@inheritDoc}
133         * @param useNames
134         * @param binding
135         *            @return <code>true</code> if matching components were found or
136         *            an empty collective type is allowed
137         */
138        public Resolver resolve(final PicoContainer container, final ComponentAdapter<?> forAdapter,
139                        final ComponentAdapter<?> injecteeAdapter, final Type expectedType, final NameBinding expectedNameBinding,
140                        final boolean useNames, final Annotation binding) {
141                final Class collectionType = getCollectionType(expectedType);
142                if (collectionType != null) {
143                        final Map<Object, ComponentAdapter<?>> componentAdapters = getMatchingComponentAdapters(container,
144                                        forAdapter, keyType, getValueType(expectedType));
145                        return new Resolver() {
146                                public boolean isResolved() {
147                                        return emptyCollection || componentAdapters.size() > 0;
148                                }
149
150                                public Object resolveInstance(final Type into) {
151                                        Object result = null;
152                                        if (collectionType.isArray()) {
153                                                result = getArrayInstance(container, collectionType, componentAdapters, into);
154                                        } else if (Map.class.isAssignableFrom(collectionType)) {
155                                                result = getMapInstance(container, collectionType, componentAdapters, into);
156                                        } else if (Collection.class.isAssignableFrom(collectionType)) {
157                                                result = getCollectionInstance(container, collectionType, componentAdapters,
158                                                                expectedNameBinding, useNames, into);
159                                        } else {
160                                                throw new PicoCompositionException(expectedType + " is not a collective type");
161                                        }
162                                        return result;
163                                }
164
165                                public ComponentAdapter<?> getComponentAdapter() {
166                                        return null;
167                                }
168                        };
169                }
170                return new Parameter.NotResolved();
171        }
172
173        public Class getCollectionType(final Type expectedType) {
174                if (expectedType instanceof Class) {
175                        return getCollectionType((Class) expectedType);
176                } else if (expectedType instanceof ParameterizedType) {
177                        final ParameterizedType type = (ParameterizedType) expectedType;
178
179                        return getCollectionType(type.getRawType());
180                } else if (expectedType instanceof GenericArrayType) {
181//                      return getCollectionType(((GenericArrayType) expectedType).getGenericComponentType());
182                  GenericArrayType type = (GenericArrayType) expectedType;
183                  Class baseType = getGenericArrayBaseType(type.getGenericComponentType());
184                  return Array.newInstance(baseType, 0).getClass();
185            }
186
187                throw new IllegalArgumentException("Unable to get collection type from " + expectedType);
188        }
189
190           private Class getGenericArrayBaseType(final Type expectedType) {
191                if (expectedType instanceof Class) {
192                    Class type = (Class) expectedType;
193                    return type;
194                }
195                else if (expectedType instanceof ParameterizedType) {
196                    ParameterizedType type = (ParameterizedType) expectedType;
197                    return getGenericArrayBaseType(type.getRawType());
198                }
199
200                throw new IllegalArgumentException("Unable to get collection type from " + expectedType);
201            }

203        /**
204         * Verify a successful dependency resolution of the parameter for the
205         * expected type. The method will only return if the expected type is one of
206         * the collection types {@link Array}, {@link Collection}or {@link Map}. An
207         * empty collection is only a valid resolution, if the
208         * <code>emptyCollection</code> flag was set.
209         *
210         * @param container
211         *            {@inheritDoc}
212         * @param adapter
213         *            {@inheritDoc}
214         * @param expectedType
215         *            {@inheritDoc}
216         * @param expectedNameBinding
217         *            {@inheritDoc}
218         * @param useNames
219         * @param binding
220         * @throws PicoCompositionException
221         *             {@inheritDoc}
222         */
223        public void verify(final PicoContainer container, final ComponentAdapter<?> adapter, final Type expectedType,
224                        final NameBinding expectedNameBinding, final boolean useNames, final Annotation binding) {
225                final Class collectionType = getCollectionType(expectedType);
226                if (collectionType != null) {
227                        final Generic<?> valueType = getValueType(expectedType);
228                        final Collection componentAdapters = getMatchingComponentAdapters(container, adapter, keyType, valueType)
229                                        .values();
230                        if (componentAdapters.isEmpty()) {
231                                if (!emptyCollection) {
232                                        throw new PicoCompositionException(expectedType + " not resolvable, no components of type "
233                                                        + valueType.toString() + " available");
234                                }
235                        } else {
236                                for (final Object componentAdapter1 : componentAdapters) {
237                                        final ComponentAdapter componentAdapter = (ComponentAdapter) componentAdapter1;
238                                        componentAdapter.verify(container);
239                                }
240                        }
241                } else {
242                        throw new PicoCompositionException(expectedType + " is not a collective type");
243                }
244        }

259         * @param adapter
260         *            a <code>ComponentAdapter</code> value

264                return adapter != null; // use parameter, prevent compiler warning

267        /**
268         * Collect the matching ComponentAdapter instances.
269         *
270         * @param container
271         *            container to use for dependency resolution
272         * @param adapter
273         *            {@link com.picocontainer.ComponentAdapter} to exclude
274         * @param keyType
275         *            the compatible type of the key
276         * @param valueType
277         *            the compatible type of the addComponent
278         * @return a {@link Map} with the ComponentAdapter instances and their
279         *         component keys as map key.
280         */
281        @SuppressWarnings({ "unchecked" })
282        protected Map<Object, ComponentAdapter<?>> getMatchingComponentAdapters(final PicoContainer container,
283                        final ComponentAdapter adapter, final Class keyType, final Generic<?> valueType) {
284                final Map<Object, ComponentAdapter<?>> adapterMap = new LinkedHashMap<Object, ComponentAdapter<?>>();
285                final PicoContainer parent = container.getParent();
286                if (parent != null) {
287                        adapterMap.putAll(getMatchingComponentAdapters(parent, adapter, keyType, valueType));
288                }
289                final Collection<ComponentAdapter<?>> allAdapters = container.getComponentAdapters();
290                for (final ComponentAdapter componentAdapter : allAdapters) {
291                        adapterMap.remove(componentAdapter.getComponentKey());
292                }
293                final List<ComponentAdapter> adapterList = List.class.cast(container.getComponentAdapters(valueType));
294                for (final ComponentAdapter componentAdapter : adapterList) {
295                        final Object key = componentAdapter.getComponentKey();
296                        if (adapter != null && key.equals(adapter.getComponentKey())) {
297                                continue;
298                        }
299                        if (keyType.isAssignableFrom(key.getClass()) && evaluate(componentAdapter)) {
300                                adapterMap.put(key, componentAdapter);
301                        }
302                }
303                return adapterMap;
304        }

306        private Class getCollectionType(final Class collectionType) {
307                if (collectionType.isArray() || Map.class.isAssignableFrom(collectionType)
308                                || Collection.class.isAssignableFrom(collectionType)) {
309                        return collectionType;
310                }
311
312                return null;
313        }

315        private Generic<?> getValueType(final Type collectionType) {
316                if (collectionType instanceof Class) {
317                        return getValueType((Class) collectionType);
318                } else if (collectionType instanceof ParameterizedType) {
319                        return getValueType((ParameterizedType) collectionType);
320                } else if (collectionType instanceof GenericArrayType) {
321            GenericArrayType genericArrayType = (GenericArrayType) collectionType;
322
323            return Generic.get(getGenericArrayBaseType(genericArrayType.getGenericComponentType()));
324        }
325                throw new IllegalArgumentException("Unable to determine collection type from " + collectionType);
326        }
327
328        private Generic<?> getValueType(final Class collectionType) {
329                Generic<?> valueType = componentValueType;
330                if (collectionType.isArray()) {
331                        valueType = Generic.get(collectionType.getComponentType());
332                }
333                return valueType;
334        }
335
336        private Generic<?> getValueType(final ParameterizedType collectionType) {
337                Generic<?> valueType = componentValueType;
338                if (Collection.class.isAssignableFrom((Class<?>) collectionType.getRawType())) {
339                        final Type type = collectionType.getActualTypeArguments()[0];
340                        if (type instanceof Class) {
341                                if (JTypeHelper.isAssignableTo(valueType, (Class) type)) {
342                                        return valueType;
343                                }
344                                valueType = Generic.get((Class) type);
345                        }
346                }
347                return valueType;
348        }

350        private Object[] getArrayInstance(final PicoContainer container, final Class expectedType,
351                        final Map<Object, ComponentAdapter<?>> adapterList, final Type into) {
352                final Object[] result = (Object[]) Array.newInstance(expectedType.getComponentType(), adapterList.size());
353                int i = 0;
354                for (final ComponentAdapter componentAdapter : adapterList.values()) {
355                        result[i] = container.getComponentInto(componentAdapter.getComponentKey(), into);
356                        i++;
357                }
358                return result;
359        }

361        @SuppressWarnings({ "unchecked" })
362        private Collection getCollectionInstance(final PicoContainer container,
363                        final Class<? extends Collection> expectedType, final Map<Object, ComponentAdapter<?>> adapterList,
364                        final NameBinding expectedNameBinding, final boolean useNames, final Type into) {
365                Class<? extends Collection> collectionType = expectedType;
366                if (collectionType.isInterface()) {
367                        // The order of tests are significant. The least generic types last.
368                        if (List.class.isAssignableFrom(collectionType)) {
369                                collectionType = ArrayList.class;
370                                // } else if
371                                // (BlockingQueue.class.isAssignableFrom(collectionType)) {
372                                // collectionType = ArrayBlockingQueue.class;
373                                // } else if (Queue.class.isAssignableFrom(collectionType)) {
374                                // collectionType = LinkedList.class;
375                        } else if (SortedSet.class.isAssignableFrom(collectionType)) {
376                                collectionType = TreeSet.class;
377                        } else if (Set.class.isAssignableFrom(collectionType)) {
378                                collectionType = HashSet.class;
379                        } else if (Collection.class.isAssignableFrom(collectionType)) {
380                                collectionType = ArrayList.class;
381                        }
382                }
383                try {
384                        final Collection result = collectionType.newInstance();
385                        for (final ComponentAdapter componentAdapter : adapterList.values()) {
386                                if (!useNames || componentAdapter.getComponentKey() == expectedNameBinding) {
387                                        result.add(container.getComponentInto(componentAdapter.getComponentKey(), into));
388                                }
389                        }
390                        return result;
391                } catch (final InstantiationException e) {
392                        // /CLOVER:OFF
393                        throw new PicoCompositionException(e);
394                        // /CLOVER:ON
395                } catch (final IllegalAccessException e) {
396                        // /CLOVER:OFF
397                        throw new PicoCompositionException(e);
398                        // /CLOVER:ON
399                }
400        }

402        @SuppressWarnings({ "unchecked" })
403        private Map getMapInstance(final PicoContainer container, final Class<? extends Map> expectedType,
404                        final Map<Object, ComponentAdapter<?>> adapterList, final Type into) {
405                Class<? extends Map> collectionType = expectedType;
406                if (collectionType.isInterface()) {
407                        // The order of tests are significant. The least generic types last.
408                        if (SortedMap.class.isAssignableFrom(collectionType)) {
409                                collectionType = TreeMap.class;
410                                // } else if
411                                // (ConcurrentMap.class.isAssignableFrom(collectionType)) {
412                                // collectionType = ConcurrentHashMap.class;
413                        } else if (Map.class.isAssignableFrom(collectionType)) {
414                                collectionType = HashMap.class;
415                        }
416                }
417                try {
418                        final Map result = collectionType.newInstance();
419                        for (final Map.Entry<Object, ComponentAdapter<?>> entry : adapterList.entrySet()) {
420                                final Object key = entry.getKey();
421                                result.put(key, container.getComponentInto(key, into));
422                        }
423                        return result;
424                } catch (final InstantiationException e) {
425                        // /CLOVER:OFF
426                        throw new PicoCompositionException(e);
427                        // /CLOVER:ON
428                } catch (final IllegalAccessException e) {
429                        // /CLOVER:OFF
430                        throw new PicoCompositionException(e);
431                        // /CLOVER:ON
432                }
433        }
