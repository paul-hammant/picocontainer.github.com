001package com.picocontainer.parameters;
002
003import java.io.Serializable;
004import java.util.Arrays;
005
006import com.picocontainer.Parameter;
007
008/***
009 * Serialization Warning.  While using a specific target type is more of an edge case, it
010 * isn't serializable and will have to be re-applied after deserialization.
011 * @author Michael Rimov
012 *
013 */
014@SuppressWarnings("serial")
015public class AccessibleObjectParameterSet implements Serializable {
016
017
018        public static final AccessibleObjectParameterSet[] EMPTY = new AccessibleObjectParameterSet[]{};
019
020        /**
021         * Can't be serialized
022         */
023        private transient Class<?> targetType;
024
025        private final Parameter[] params;
026
027        private final String name;
028
029
030        public AccessibleObjectParameterSet(final String name, final Parameter... params) {
031                this.name = name;
032                this.params = params;
033        }
034
035        public AccessibleObjectParameterSet(final Class<?> targetType, final String name, final Parameter... params) {
036                this.targetType = targetType;
037                this.name = name;
038                this.params = params;
039        }
040
041        public Class<?> getTargetType() {
042                return targetType;
043        }
044
045        public void setTargetType(final Class<?> targetType) {
046                this.targetType = targetType;
047        }
048
049        public Parameter[] getParams() {
050                return params;
051        }
052
053        public String getName() {
054                return name;
055        }
056
057
058        @Override
059        public String toString() {
060                return "AccessibleObjectParameterSet [targetType=" + targetType + ", params=" + Arrays.toString(params)
061                                + ", name=" + name + "]";
062        }
063
064}
