009package com.picocontainer;
010
011import java.lang.reflect.Type;
012
013/**
014 * Implementers are responsible for instantiating and injecting dependancies into
015 * Constructors, Methods and Fields.
016 */
017public interface Injector<T> extends ComponentAdapter<T> {
018
019    /**
020     * A preexiting component instance can be injected into after instantiation
021     *
022     *
023     * @param container the container that can provide injectable dependencies
024     * @param into
025     * @param instance the instance to
026     * @return
027     */
028    Object decorateComponentInstance(PicoContainer container, Type into, T instance);
029
030
031    /**
032     * Does a partial decoration. This is necessary since in composite injection, you need to
033     * inject the superclass fields/methods first before injecting the subtype fields/methods.
034     * @param container
035     * @param into
036     * @param instance the instance we're decorating.
037     * @param superclassPortion the portion of the object to decorate.
038     * @return
039     */
040    Object partiallyDecorateComponentInstance(PicoContainer container, Type into, T instance, Class<?> superclassPortion);
041
042}
