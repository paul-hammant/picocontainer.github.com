076            boolean useNames = AbstractBehavior.arePropertiesPresent(componentProps, Characteristics.USE_NAMES, true);
077            boolean requireConsumptionOfAllParameters = !(AbstractBehavior.arePropertiesPresent(componentProps, Characteristics.ALLOW_UNUSED_PARAMETERS, false));

082    public class MethodInjectionByReflectionMethod extends AbstractInjectionType {
083        private final Method injectionMethod;
084
085        public MethodInjectionByReflectionMethod(final Method injectionMethod) {
086            this.injectionMethod = injectionMethod;
087        }

089        public <T> ComponentAdapter<T> createComponentAdapter(final ComponentMonitor monitor, final LifecycleStrategy lifecycle, final Properties componentProps, final Object key, final Class<T> impl, final ConstructorParameters constructorParams, final FieldParameters[] fieldParams, final MethodParameters[] methodParams) throws PicoCompositionException {
090            boolean useNames = AbstractBehavior.arePropertiesPresent(componentProps, Characteristics.USE_NAMES, true);
091            boolean requireConsumptionOfAllParameters = !(AbstractBehavior.arePropertiesPresent(componentProps, Characteristics.ALLOW_UNUSED_PARAMETERS, false));
092
093            if (injectionMethod.getDeclaringClass().isAssignableFrom(impl)) {
094                return wrapLifeCycle(monitor.newInjector(new SpecificMethodInjector(key, impl, monitor, useNames, requireConsumptionOfAllParameters, methodParams, injectionMethod)), lifecycle);
095            } else {
096                throw new PicoCompositionException("method [" + injectionMethod + "] not on impl " + impl.getName());
097            }
098        }

101    /**
102     * Injection will happen through a single method for the component.
103     *
104     * Most likely it is a method called 'inject', though that can be overridden.
105     *
106     * @author Paul Hammant
107     * @author Aslak Helles&oslash;y

113    public static class MethodInjector<T> extends MultiArgMemberInjector<T> {
114        private transient ThreadLocalCyclicDependencyGuard<Object> instantiationGuard;
115        private final String methodNamePrefix;

117        /**
118         * Creates a MethodInjector
119         *
120         * @param key            the search key for this implementation
121         * @param impl the concrete implementation
122         * @param monitor                 the component monitor used by this addAdapter
123         * @param methodName              the method name
124         * @param useNames                use argument names when looking up dependencies
125         * @param parameters              the parameters to use for the initialization
126         * @throws com.picocontainer.injectors.AbstractInjector.NotConcreteRegistrationException
127         *                              if the implementation is not a concrete class.
128         * @throws NullPointerException if one of the parameters is <code>null</code>
129         */
130        public MethodInjector(final Object key, final Class<T> impl,
131                                                final ComponentMonitor monitor, final String methodName,
132                                                final boolean useNames, final boolean useAllParameters,
133                              final MethodParameters... parameters) throws NotConcreteRegistrationException {
134            super(key, impl, parameters, monitor, useNames, useAllParameters);
135            this.methodNamePrefix = methodName;
136        }

140        protected List<Method> getInjectorMethods() {
141                Class<?> toIntrospect = null;
142            //Method[] methods = new Method[0];
143            try {
144                //methods = super.getComponentImplementation().getMethods();
145                toIntrospect = super.getComponentImplementation();
146            } catch (AmbiguousComponentResolutionException e) {
147                e.setComponent(getComponentImplementation());
148                throw e;
149            }
150
151            if (toIntrospect == null) {
152                throw new NullPointerException("No implementation class defined for " + this);
153            }
154
155
156            HashMap<String, Set<Method>> allMethodsAnalyzed = new HashMap<String,Set<Method>>();
157            List<Method> methodz = new ArrayList<Method>();
158            recursiveCheckInjectorMethods(toIntrospect, toIntrospect, methodz, allMethodsAnalyzed);
159
160            //Inject in JSR330 compliant order.
161            Collections.sort(methodz, new JSRAccessibleObjectOrderComparator());
162
163            return methodz;
164        }
165
166        /**
167         * Goes through all methods (including private base classes) to look for potential injection methods.
168         * @param originalType
169         * @param type
170         * @param receiver
171         * @param allMethodsAnalyzed
172         */
173        protected void recursiveCheckInjectorMethods(final Class<?> originalType, final Class<?> type, final List<Method> receiver, final HashMap<String, Set<Method>> allMethodsAnalyzed) {
174                if (type.isAssignableFrom(Object.class)) {
175                        return;
176                }
177
178                for (Method eachMethod : type.getDeclaredMethods()) {
179                        if(alreadyAnalyzedChildClassMethod(eachMethod, allMethodsAnalyzed)) {
180                                //This method was defined in a child class, what the child class says, goes.
181                                continue;
182                        }
183
184                        addToMethodsAnalyzed(allMethodsAnalyzed, eachMethod);
185
186                        if (isInjectorMethod(originalType, eachMethod)) {
187                                receiver.add(eachMethod);
188                        }
189                }
190
191                recursiveCheckInjectorMethods(originalType, type.getSuperclass(), receiver, allMethodsAnalyzed);
192        }
193
194
195        private void addToMethodsAnalyzed(final HashMap<String, Set<Method>> allMethodsAnalyzed, final Method eachMethod) {
196                if (!allMethodsAnalyzed.containsKey(eachMethod.getName())) {
197                        allMethodsAnalyzed.put(eachMethod.getName(), new HashSet<Method>());
198                }
199
200                allMethodsAnalyzed.get(eachMethod.getName()).add(eachMethod);
201                }
202
203                private boolean alreadyAnalyzedChildClassMethod(final Method eachMethod,
204                                final HashMap<String, Set<Method>> allMethodsAnalyzed) {
205
206                        Set<Method> methodsByName = allMethodsAnalyzed.get(eachMethod.getName());
207                        if (methodsByName == null) {
208                                return false;
209                        }
210
211                        return false;
212                }
213
214                protected boolean isInjectorMethod(final Class<?> originalType, final Method method) {
215            return method.getName().startsWith(methodNamePrefix);
216        }

218                @Override
219        @SuppressWarnings("unchecked")
220        public T getComponentInstance(final PicoContainer container, final Type into) throws PicoCompositionException {
221                boolean i_Instantiated = false;
222                T result;
223                try {
224                    if (instantiationGuard == null) {
225                        i_Instantiated = true;
226                        instantiationGuard = new ThreadLocalCyclicDependencyGuard<Object>() {
227                            @Override
228                            @SuppressWarnings("synthetic-access")
229                            public Object run(final Object instance) {
230                                List<Method> methods = getInjectorMethods();
231                                T inst = null;
232                                ComponentMonitor monitor = currentMonitor();
233                                Method lastMethod = null;
234                                try {
235                                    monitor.instantiating(container, MethodInjector.this, null);
236                                    long startTime = System.currentTimeMillis();
237                                    Object[] methodParameters = null;
238                                    inst = getComponentImplementation().newInstance();
239                                    for (Method method : methods) {
240                                        lastMethod = method;
241                                        methodParameters = getMemberArguments(guardedContainer, method, into);
242                                        invokeMethod(method, methodParameters, inst, container);
243                                    }
244                                    monitor.instantiated(container, MethodInjector.this,
245                                                                  null, inst, methodParameters, System.currentTimeMillis() - startTime);
246                                    return inst;
247                                } catch (InstantiationException e) {
248                                    return caughtInstantiationException(monitor, null, e, container);
249                                } catch (IllegalAccessException e) {
250                                    return caughtIllegalAccessException(monitor, lastMethod, inst, e);
251
252                                }
253                            }
254                        };
255                    }
256                    instantiationGuard.setGuardedContainer(container);
257                    result =  (T) instantiationGuard.observe(getComponentImplementation(), null);
258                } finally {
259                    if (i_Instantiated) {
260                        instantiationGuard.remove();
261                        instantiationGuard = null;
262                    }
263                }
264            return result;
265        }

278                @Override
279                @SuppressWarnings("unchecked")
280                public Object partiallyDecorateComponentInstance(final PicoContainer container, final Type into, final T instance,
281                                final Class<?> injectionTypeFilter) {
282                        boolean iInstantiated = false;
283                        Object o;
284                        try {
285                    if (instantiationGuard == null) {
286                        iInstantiated = true;
287                        instantiationGuard = new ThreadLocalCyclicDependencyGuard<Object>() {
288                            @Override
289                            @SuppressWarnings("synthetic-access")
290                            public Object run(final Object inst) {
291                                List<Method> methods = getInjectorMethods();
292                                Object lastReturn = null;
293                                for (Method method : methods) {
294                                        Class<?> methodClass = method.getDeclaringClass();
295                                        Class<?> filterClass = injectionTypeFilter;
296                                        if (!allowedMethodBasedOnFilter(filterClass, method)) {
297                                                continue;
298                                        }
299
300                                    if (methodClass.isAssignableFrom(inst.getClass())) {
301                                        Object[] methodParameters = getMemberArguments(guardedContainer, method, into);
302                                        lastReturn = invokeMethod(method, methodParameters, (T) inst, container);
303                                    }
304                                }
305                                return lastReturn;
306                            }
307
308                        };
309                    }
310                    instantiationGuard.setGuardedContainer(container);
311                    o = instantiationGuard.observe(getComponentImplementation(), instance);
312                        } finally {
313                    if (iInstantiated) {
314                        instantiationGuard.remove();
315                        instantiationGuard = null;
316                    }
317                        }
318
319            return o;
320        }

333        Object invokeMethod(final Method method, final Object[] methodParameters, final T instance, final PicoContainer container) {
334            try {
335                Object rv = currentMonitor().invoking(container, MethodInjector.this, (Member) method, instance, methodParameters);
336                if (rv == ComponentMonitor.KEEP) {
337                    long str = System.currentTimeMillis();
338                    makeAccessibleIfDesired(method);
339                    rv = method.invoke(instance, methodParameters);
340                    currentMonitor().invoked(container, MethodInjector.this, method, instance, System.currentTimeMillis() - str, rv, methodParameters);
341                }
342                return rv;
343            } catch (IllegalAccessException e) {
344                return caughtIllegalAccessException(currentMonitor(), method, instance, e);
345            } catch (InvocationTargetException e) {
346                currentMonitor().invocationFailed(method, instance, e);
347                if (e.getTargetException() instanceof RuntimeException) {
348                    throw (RuntimeException) e.getTargetException();
349                } else if (e.getTargetException() instanceof Error) {
350                    throw (Error) e.getTargetException();
351                }
352                throw new PicoCompositionException(e);
353            }
354        }

366                @Override
367                @SuppressWarnings("unchecked")
368        public void verify(final PicoContainer container) throws PicoCompositionException {
369                        boolean i_created = false;
370                        try {
371                    if (verifyingGuard == null) {
372                        i_created = true;
373                        verifyingGuard = new ThreadLocalCyclicDependencyGuard<Void>() {
374                            @Override
375                            public Void run(final Object inst) {
376                                final List<Method> methods = getInjectorMethods();
377                                for (Method method : methods) {
378
379                                    final Class[] parameterTypes = method.getParameterTypes();
380
381                                    AccessibleObjectParameterSet paramsForMethod = getParameterToUseForObject(method, parameters);
382
383
384                                    final Parameter[] currentParameters = paramsForMethod != null ? paramsForMethod.getParams() : createDefaultParameters(parameterTypes.length);
385                                    for (int i = 0; i < currentParameters.length; i++) {
386                                        currentParameters[i].verify(container, MethodInjector.this, parameterTypes[i],
387                                                new ParameterNameBinding(getParanamer(), method, i), useNames(),
388                                                getBindings(method.getParameterAnnotations())[i]);
389                                    }
390
391                                }
392                                return null;
393                            }
394                        };
395                    }
396                    verifyingGuard.setGuardedContainer(container);
397                    verifyingGuard.observe(getComponentImplementation(), null);
398                        } finally {
399                    if (i_created) {
400                        verifyingGuard.remove();
401                        verifyingGuard = null;
402                    }
403                        }
404        }

408        @Override
409        public String getDescriptor() {
410            StringBuilder mthds = new StringBuilder();
411            for (Method method : getInjectorMethods()) {
412                mthds.append(",").append(method.getName());
413            }
414            return "MethodInjector["+mthds.substring(1)+"]-";
415        }

417        @Override
418        protected boolean isNullParamAllowed(final AccessibleObject member, final int i) {
419            Annotation[] annotations = ((Method) member).getParameterAnnotations()[i];
420            for (Annotation annotation : annotations) {
421                if (annotation instanceof Nullable) {
422                    return true;
423                }
424            }
425            return false;
426        }
