103        Object retVal = AccessController.doPrivileged(new PrivilegedAction<Object>() {
104            public Object run() {
105                try {
106                    return getComponentImplementation().getConstructor((Class[])null);
107                } catch (NoSuchMethodException e) {
108                    return new PicoCompositionException(e);
109                } catch (SecurityException e) {
110                    return new PicoCompositionException(e);
111                }
112            }
113        });
114        if (retVal instanceof Constructor) {
115            return (Constructor<?>) retVal;
116        } else {
117            throw (PicoCompositionException) retVal;

160    ParameterToAccessibleObjectPair[] getMatchingParameterListForMembers(final PicoContainer container) throws PicoCompositionException {
161        if (initialized == false) {
162                synchronized(this) {
163                        if (initialized == false) {
164                                initializeInjectionMembersAndTypeLists();
165                        }
166                }
167        }
168
169        final List<Object> matchingParameterList = new ArrayList<Object>(Collections.nCopies(injectionMembers.size(), null));
170
171        final Set<AccessibleObjectParameterSet> notMatchingParameters = matchParameters(container, matchingParameterList, parameters);
172
173        final Set<Type> unsatisfiableDependencyTypes = new HashSet<Type>();
174        final List<AccessibleObject> unsatisfiableDependencyMembers = new ArrayList<AccessibleObject>();
175
176        for (int i = 0; i < matchingParameterList.size(); i++) {
177                ParameterToAccessibleObjectPair param = (ParameterToAccessibleObjectPair)matchingParameterList.get(i);
178            if (param == null ||  !param.isResolved()) {
179                unsatisfiableDependencyTypes.add(injectionTypes[i]);
180                unsatisfiableDependencyMembers.add(injectionMembers.get(i));
181            }
182        }
183        if (unsatisfiableDependencyTypes.size() > 0) {
184                unsatisfiedDependencies(container, unsatisfiableDependencyTypes, unsatisfiableDependencyMembers);
185        } else if (notMatchingParameters.size() > 0 && this.requireConsumptionOfAllParameters) {
186            throw new PicoCompositionException("Following parameters do not match any of the injectionMembers for " + getComponentImplementation() + ": " + notMatchingParameters.toString());
187        }
188        return matchingParameterList.toArray(new ParameterToAccessibleObjectPair[matchingParameterList.size()]);
189    }

194        /**
195     * Returns a set of integers that point to where in the Parameter array unmatched parameters exist.
196     * @param container
197     * @param matchingParameterList
198     * @param assignedParameters {@link com.picocontainer.Parameter} for the current object being instantiated.
199     * @return set of integers pointing to the index in the parameter array things went awry.
200     */
201    private Set<AccessibleObjectParameterSet> matchParameters(final PicoContainer container, final List<Object> matchingParameterList, final AccessibleObjectParameterSet... assignedParameters) {
202
203        Set<AccessibleObjectParameterSet> unmatchedParameters = new HashSet<AccessibleObjectParameterSet>();
204
205        for (AccessibleObject eachObject : injectionMembers) {
206                AccessibleObjectParameterSet currentParameter =   getParameterToUseForObject(eachObject, assignedParameters);
207
208                if (currentParameter == null) {
209                        currentParameter = this.constructAccessibleObjectParameterSet(eachObject,new Parameter[] {constructDefaultComponentParameter()});
210                }
211
212                if (!matchParameter(container, matchingParameterList, currentParameter)) {
213                        unmatchedParameters.add(currentParameter);
214                }
215        }
216
217        return unmatchedParameters;
218    }

220    private boolean matchParameter(final PicoContainer container, final List<Object> matchingParameterList, final AccessibleObjectParameterSet parameter) {
221        for (int j = 0; j < injectionTypes.length; j++) {
222
223
224            Object o = matchingParameterList.get(j);
225            AccessibleObject targetInjectionMember = getTargetInjectionMember(injectionMembers, j, parameter.getParams()[0]);
226            if (targetInjectionMember == null) {
227                return false;
228            }
229
230            AccessibleObjectParameterSet paramToUse = getParameterToUseForObject(targetInjectionMember, parameter);
231            if (paramToUse == null) {
232                paramToUse = constructAccessibleObjectParameterSet(targetInjectionMember);
233            }
234
235            try {
236                if (o == null
237                        && paramToUse.getParams()[0].resolve(container, this, null, injectionTypes[j],
238                                                   makeParameterNameImpl(targetInjectionMember),
239                                                   useNames(), bindings[j]).isResolved()) {
240                    matchingParameterList.set(j, new ParameterToAccessibleObjectPair(targetInjectionMember, paramToUse));
241                    return true;
242                }
243            } catch (AmbiguousComponentResolutionException e) {
244                e.setComponent(getComponentImplementation());
245                e.setMember(injectionMembers.get(j));
246                throw e;
247            }
248        }
249        return false;
250    }

256        /**
257         * Retrieves the appropriate injection member or null if the parameter doesn't match anything we know about and {@linkplain #requireConsumptionOfAllParameters}
258         * is set to false.
259         * @param injectionMembers
260         * @param currentIndex
261         * @param parameter
262         * @return Might return null if the parameter doesn't apply to this target.
263         */
264        private AccessibleObject getTargetInjectionMember(final List<AccessibleObject> injectionMembers, final int currentIndex,
265                        final Parameter parameter) {
266
267                if (parameter.getTargetName() == null) {
268                        return injectionMembers.get(currentIndex);
269                }
270
271                for (AccessibleObject eachObject : injectionMembers) {
272                        if (isAccessibleObjectEqualToParameterTarget(eachObject, parameter)) {
273                                return eachObject;
274                        }
275                }
276
277                if (this.requireConsumptionOfAllParameters) {
278                        throw new PicoCompositionException("There was no matching target field/method for target name "
279                                                + parameter.getTargetName()
280                                                + " using injector " + this.getDescriptor());
281                }
282
283                return null;
284        }

286        protected NameBinding makeParameterNameImpl(final AccessibleObject member) {
287                if (member == null) {
288                        throw new NullPointerException("member");
289                }
290
291        if (paranamer == null) {
292            paranamer = new CachingParanamer(new AnnotationParanamer(new AdaptiveParanamer()));
293        }
294        return new ParameterNameBinding(paranamer,  member, 0);

299    @Override
300        public T getComponentInstance(final PicoContainer container, final Type into) throws PicoCompositionException {
301        final Constructor<?> constructor = getConstructor();
302        boolean iInstantiated = false;
303        T result;
304        try {
305                if (instantiationGuard == null) {
306                        iInstantiated = true;
307                    instantiationGuard = new ThreadLocalCyclicDependencyGuard<T>() {
308                        @Override
309                                        public T run(final Object instance) {
310                            final ParameterToAccessibleObjectPair[] matchingParameters = getMatchingParameterListForMembers(guardedContainer);
311                            Object componentInstance = makeInstance(container, constructor, currentMonitor());
312                            return  decorateComponentInstance(matchingParameters, currentMonitor(), componentInstance, container, guardedContainer, into, null);
313                        }
314                    };
315                }
316                instantiationGuard.setGuardedContainer(container);
317                result = instantiationGuard.observe(getComponentImplementation(), null);
318        } finally {
319                if (iInstantiated) {
320                        instantiationGuard.remove();
321                        instantiationGuard = null;
322                }
323        }
324        return result;
325    }

327    T decorateComponentInstance(final ParameterToAccessibleObjectPair[] matchingParameters, final ComponentMonitor monitor, final Object componentInstance, final PicoContainer container, final PicoContainer guardedContainer, final Type into, final Class<?> partialDecorationFilter) {
328        AccessibleObject member = null;
329        Object injected[] = new Object[injectionMembers.size()];
330        Object lastReturn = null;
331        try {
332            for (int i = 0; i < matchingParameters.length; i++) {
333                if (matchingParameters[i] != null) {
334                        member = matchingParameters[i].getAccessibleObject();
335                }
336
337
338                //Skip it, we're only doing a partial injection
339                if (partialDecorationFilter != null && !partialDecorationFilter.equals( ((Member)member).getDeclaringClass() )) {
340                        continue;
341                }
342
343                if (matchingParameters[i] != null && matchingParameters[i].isResolved()) {
344                        //Again, interative injector only supports 1 parameter
345                        //per method to inject.
346                    Object toInject = matchingParameters[i].getAccessibleObjectParameters().getParams()[0].resolve(guardedContainer, this, null, injectionTypes[i],
347                                                                            makeParameterNameImpl(injectionMembers.get(i)),
348                                                                            useNames(), bindings[i]).resolveInstance(into);
349                    Object rv = monitor.invoking(container, this, (Member) member, componentInstance, new Object[] {toInject});
350                    if (rv == ComponentMonitor.KEEP) {
351                        long str = System.currentTimeMillis();
352                        lastReturn = injectIntoMember(member, componentInstance, toInject);
353                        monitor.invoked(container, this, (Member) member, componentInstance, System.currentTimeMillis() - str, lastReturn, new Object[] {toInject});
354                    } else {
355                        lastReturn = rv;
356                    }
357                    injected[i] = toInject;
358                }
359            }
360            return (T) memberInvocationReturn(lastReturn, member, componentInstance);
361        } catch (InvocationTargetException e) {
362            return caughtInvocationTargetException(monitor, (Member) member, componentInstance, e);
363        } catch (IllegalAccessException e) {
364            return caughtIllegalAccessException(monitor, (Member) member, componentInstance, e);
365        }
366    }

370    private Object makeInstance(final PicoContainer container, final Constructor constructor, final ComponentMonitor monitor) {
371        long startTime = System.currentTimeMillis();
372        Constructor constructorToUse = monitor.instantiating(container,
373                                                                      IterativeInjector.this, constructor);
374        Object componentInstance;
375        try {
376            componentInstance = newInstance(constructorToUse, null);
377        } catch (InvocationTargetException e) {
378            monitor.instantiationFailed(container, IterativeInjector.this, constructorToUse, e);
379            if (e.getTargetException() instanceof RuntimeException) {
380                throw (RuntimeException)e.getTargetException();
381            } else if (e.getTargetException() instanceof Error) {
382                throw (Error)e.getTargetException();
383            }
384            throw new PicoCompositionException(e.getTargetException());
385        } catch (InstantiationException e) {
386            return caughtInstantiationException(monitor, constructor, e, container);
387        } catch (IllegalAccessException e) {
388            return caughtIllegalAccessException(monitor, constructor, e, container);
389        }
390        monitor.instantiated(container,
391                                      IterativeInjector.this,
392                                      constructorToUse,
393                                      componentInstance,
394                                      NONE,
395                                      System.currentTimeMillis() - startTime);
396        return componentInstance;
397    }

406    public Object partiallyDecorateComponentInstance(final PicoContainer container, final Type into, final T instance, final Class<?> superclassPortion) {
407        boolean iInstantiated = false;
408        T result;
409        try {
410                if (instantiationGuard == null) {
411                        iInstantiated = true;
412                    instantiationGuard = new ThreadLocalCyclicDependencyGuard<T>() {
413                        @Override
414                                        public T run(final Object inst) {
415                            final ParameterToAccessibleObjectPair[] matchingParameters = getMatchingParameterListForMembers(guardedContainer);
416                            return decorateComponentInstance(matchingParameters, currentMonitor(), inst, container, guardedContainer, into, superclassPortion);
417                        }
418                    };
419                }
420                instantiationGuard.setGuardedContainer(container);
421                result =  instantiationGuard.observe(getComponentImplementation(), instance);
422        } finally {
423                if (iInstantiated) {
424                        instantiationGuard.remove();
425                        instantiationGuard = null;
426                }
427        }
428        return result;
429    }

464    protected void initializeInjectionMembersAndTypeLists() {
465        injectionMembers = new ArrayList<AccessibleObject>();
466        Set<String> injectionMemberNames = new HashSet<String>();
467        List<Annotation> bingingIds = new ArrayList<Annotation>();
468        final List<String> nameList = new ArrayList<String>();
469        final List<Type> typeList = new ArrayList<Type>();
470        final Method[] methods = getMethods();
471        for (final Method method : methods) {
472            final Type[] parameterTypes = method.getGenericParameterTypes();
473            fixGenericParameterTypes(method, parameterTypes);
474
475            String methodSignature = crudeMethodSignature(method);
476
477            // We're only interested if there is only one parameter and the method name is bean-style.
478            if (parameterTypes.length == 1) {
479                boolean isInjector = isInjectorMethod(method);
480                // ... and the method name is bean-style.
481                // We're also not interested in dupes from parent classes (not all JDK impls)
482                if (isInjector && !injectionMemberNames.contains(methodSignature)) {
483                    injectionMembers.add(method);
484                    injectionMemberNames.add(methodSignature);
485                    nameList.add(getName(method));
486                    typeList.add(box(parameterTypes[0]));
487                    bingingIds.add(getBindings(method, 0));
488                }
489            }
490        }
491        injectionTypes = typeList.toArray(new Type[0]);
492        bindings = bingingIds.toArray(new Annotation[0]);
493        initialized = true;
494    }

496    public static String crudeMethodSignature(final Method method) {
497        StringBuilder sb = new StringBuilder();
498        sb.append(method.getReturnType().getName());
499        sb.append(method.getName());
500        for (Class<?> pType : method.getParameterTypes()) {
501            sb.append(pType.getName());
502        }
503        return sb.toString();
504    }

511    private void fixGenericParameterTypes(final Method method, final Type[] parameterTypes) {
512        for (int i = 0; i < parameterTypes.length; i++) {
513            Type parameterType = parameterTypes[i];
514            if (parameterType instanceof TypeVariable) {
515                parameterTypes[i] = method.getParameterTypes()[i];
516            }
517        }
