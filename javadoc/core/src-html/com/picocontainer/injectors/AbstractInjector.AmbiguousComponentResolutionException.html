062    /**
063     * Constructs a new ComponentAdapter for the given key and implementation.
064     * @param key the search key for this implementation
065     * @param impl the concrete implementation
066     * @param monitor the component monitor used by this ComponentAdapter
067     * @param parameters the parameters to use for the initialization
068     * @throws com.picocontainer.injectors.AbstractInjector.NotConcreteRegistrationException if the implementation is not a concrete class
069     * @throws NullPointerException if one of the parameters is <code>null</code>
070     */
071    protected AbstractInjector(final Object key, final Class<?> impl,
072                               final ComponentMonitor monitor, final boolean useNames,
073                               final AccessibleObjectParameterSet... parameters) {
074        super(key, impl, monitor);
075        this.useNames = useNames;
076        checkConcrete();
077        if (parameters != null) {
078            for (int i = 0; i < parameters.length; i++) {
079                if(parameters[i] == null) {
080                    throw new NullPointerException("Parameter " + i + " is null");
081                }
082
083                if (parameters[i].getParams() != null) {
084                        Parameter[] nestedParams = parameters[i].getParams();
085                        for (int j= 0; j < nestedParams.length; j++) {
086                                if (nestedParams[j] == null){
087                                        throw new NullPointerException("Parameter " + j + " inside " + parameters[i] + " is null" );
088                                }
089                        }
090                }
091            }
092        }
093        this.parameters = parameters;
094    }

109    private void checkConcrete() throws NotConcreteRegistrationException {
110        // Assert that the component class is concrete.
111        boolean isAbstract = (getComponentImplementation().getModifiers() & Modifier.ABSTRACT) == Modifier.ABSTRACT;
112        if (getComponentImplementation().isInterface() || isAbstract) {
113            throw new NotConcreteRegistrationException(getComponentImplementation());
114        }

117    protected Parameter[] createDefaultParameters(final AccessibleObject member) {
118        int length = 0;
119        if (member instanceof Constructor) {
120                length = ((Constructor<?>)member).getParameterTypes().length;
121        } else if (member instanceof Field) {
122                length = 1;
123        } else if (member instanceof Method) {
124                length = ((Method)member).getParameterTypes().length;
125        } else {
126                throwUnknownAccessibleObjectType(member);
127        }
128
129        return createDefaultParameters(length);
130
131    }

161    /**
162     * Constructs an appropriate {@link com.picocontainer.parameters.AccessibleObjectParameterSet} based on the type
163     * of {@link java.lang.reflect.AccessibleObject} sent.  If params are null or zero length then default parameter is used.
164     * @param object
165     * @param params
166     * @return
167     */
168    protected AccessibleObjectParameterSet constructAccessibleObjectParameterSet(final AccessibleObject object, final Parameter... params) {
169        if (object instanceof Constructor) {
170                return new ConstructorParameters(params);
171        } else if (object instanceof Field) {
172                return new FieldParameters( ((Field)object).getDeclaringClass(),  ((Field) object).getName(), params);
173        } else if (object instanceof Method) {
174                return new MethodParameters( ((Method)object).getDeclaringClass(),  ((Method) object).getName(), params);
175        }  else {
176                throwUnknownAccessibleObjectType(object);
177                //Never gets here
178                return null;
179        }
180    }

185        protected Parameter[] createDefaultParamsBasedOnTypeOfAccessibleObject(final AccessibleObject object) {
186        if (object instanceof Constructor) {
187                return createDefaultParameters( ((Constructor)object).getParameterTypes().length );
188        }
189
190        if (object instanceof Field) {
191                return createDefaultParameters(1);
192        }
193
194        if (object instanceof Method) {
195                return createDefaultParameters( ((Method)object).getParameterTypes().length );
196        }
197
198        throwUnknownAccessibleObjectType(object);
199                //Never gets here
200                return null;

215        /**
216     * @param targetInjectionMember
217     * @param assignedParameters
218     * @return null if no parameter set for the given accessible object has been defined.
219     */
220    protected final AccessibleObjectParameterSet getParameterToUseForObject(final AccessibleObject targetInjectionMember, final AccessibleObjectParameterSet... assignedParameters) {
221        if (assignedParameters == null || assignedParameters.length == 0) {
222                Parameter[] paramsToUse = this.createDefaultParameters(targetInjectionMember);
223                paramsToUse = this.interceptParametersToUse(paramsToUse, targetInjectionMember);
224                return this.constructAccessibleObjectParameterSet(targetInjectionMember, paramsToUse);
225        }
226
227
228        for (AccessibleObjectParameterSet eachParameterSet : assignedParameters) {
229
230                //if a target type is defined then we have to match against it.
231                //This allows injection into private members of base classes of the same name
232                //as the subclasses.
233                Class<?> targetType = eachParameterSet.getTargetType();
234                if (targetType != null && !targetType.equals(getAccessibleObjectDefiningType(targetInjectionMember))) {
235                        continue;
236                }
237
238                if (eachParameterSet.getName().equals(getAccessibleObjectName(targetInjectionMember))) {
239                        //Allow parmeter substitution
240                Parameter[] paramsToUse = this.interceptParametersToUse(eachParameterSet.getParams(), targetInjectionMember);
241                        return constructAccessibleObjectParameterSet(targetInjectionMember, paramsToUse);
242                }
243        }
244
245        Parameter[] paramsToUse = this.createDefaultParameters(targetInjectionMember);
246        paramsToUse = this.interceptParametersToUse(paramsToUse, targetInjectionMember);
247                return this.constructAccessibleObjectParameterSet(targetInjectionMember, paramsToUse);
248        }

253        /**
254     * Retrieves the enclosing class of the accessible object. (Constructor, Method, and Field all
255     * supply the method "getDeclaringClass()", yet it isn't supplied in the AccessibleObject base class.
256     * @param targetAccessibleObject
257     * @return the enclosing type of the accessible object.
258     */
259    protected Class<?> getAccessibleObjectDefiningType(final AccessibleObject targetAccessibleObject) {
260        if (targetAccessibleObject == null) {
261                throw new NullPointerException("targetAccessibleObject");
262        }
263
264        try {
265                        Method declaringClassMethod = targetAccessibleObject.getClass().getMethod("getDeclaringClass");
266                        return (Class<?>) declaringClassMethod.invoke(targetAccessibleObject);
267                } catch (NoSuchMethodException e) {
268                        throw new PicoCompositionException("Target Type '" + targetAccessibleObject.getClass()
269                                        + "' does not appear to suppot getDeclaringClass().  Please override getAccessibleObjectDefiningType() for your type of injection", e);
270                } catch (IllegalAccessException e) {
271                        throw new PicoCompositionException("Error invoking 'getDeclaringClass()' in type "
272                + targetAccessibleObject.getClass(), e);
273                } catch (InvocationTargetException e) {
274                        throw new PicoCompositionException("Error invoking 'getDeclaringClass()' in type "
275                + targetAccessibleObject.getClass(), e);
276                }
277
278    }

280    /**
281     * Retrieves the name of the accessible object or null if it doesn't have one (such as a constructor)
282     * @param targetAccessibleObject
283     * @return
284     */
285    public String getAccessibleObjectName(final AccessibleObject targetAccessibleObject) {
286        if (targetAccessibleObject == null) {
287                throw new NullPointerException("targetAccessibleObject");
288        }
289
290        if (targetAccessibleObject instanceof Constructor) {
291                return null;
292        }
293
294        try {
295                        Method declaringClassMethod = targetAccessibleObject.getClass().getMethod("getName");
296                        return (String) declaringClassMethod.invoke(targetAccessibleObject);
297                } catch (NoSuchMethodException e) {
298                        throw new PicoCompositionException("Target Type '" + targetAccessibleObject.getClass()
299                                        + "' does not appear to suppot getName().  Please override getAccessibleObjectDefiningType() for your type of injection", e);
300                } catch (IllegalAccessException e) {
301                        throw new PicoCompositionException("Error invoking 'getName()' in type "
302                + targetAccessibleObject.getClass(), e);
303                } catch (InvocationTargetException e) {
304                        throw new PicoCompositionException("Error invoking 'getName()' in type "
305                + targetAccessibleObject.getClass(), e);
306                }
307    }

322    @Override
323        public void accept(final PicoVisitor visitor) {
324        super.accept(visitor);
325        if (parameters != null) {
326            for (AccessibleObjectParameterSet parameter : parameters) {
327                if (parameter.getParams() != null) {
328                        for (Parameter eachContainedParameter : parameter.getParams()) {
329                                eachContainedParameter.accept(visitor);
330                        }
331                }
332            }
333        }

344     * @param constructor the constructor to use

351    protected T newInstance(final Constructor<T> constructor, final Object[] parameters)
352            throws InstantiationException, IllegalAccessException, InvocationTargetException {
353        try {
354                        return constructor.newInstance(parameters);
355                } catch (IllegalArgumentException e) {
356                        //Chain it with the calling parameters to give us some real information.
357                        throw new IllegalArgumentException("Type mismatch calling constructor "
358                                                + constructor
359                                                + " with parameters "
360                                                + Arrays.deepToString(parameters), e);
361                }

424    @SuppressWarnings("rawtypes")
425        protected Type box(final Type parameterType) {
426        if (parameterType instanceof Class && ((Class) parameterType).isPrimitive()) {
427            String parameterTypeName = ((Class) parameterType).getName();
428            if (parameterTypeName == "int") {
429                return Integer.class;
430            } else if (parameterTypeName == "boolean") {
431                return Boolean.class;
432            } else if (parameterTypeName == "long") {
433                return Long.class;
434            } else if (parameterTypeName == "float") {
435                return Float.class;
436            } else if (parameterTypeName == "double") {
437                return Double.class;
438            } else if (parameterTypeName == "char") {
439                return Character.class;
440            } else if (parameterTypeName == "byte") {
441                return Byte.class;
442            } else if (parameterTypeName == "short") {
443                return Short.class;
444            }
445        }
446        return parameterType;
447    }

450        /**
451         * Retured true if all fields are static members, throws
452         * @param showsIsStaticInjection
453         * @param fieldsToInject list of fields/methods to be injected
454         * @return true if all members are static
455         * @throws PicoCompositionException if fieldsToInject has a mix of static and non static
456         * members.
457         */
458        protected boolean isStaticInjection(final Member... fieldsToInject) {
459                Boolean isStaticFields = null;
460        for (Member eachField : fieldsToInject) {
461                if (Modifier.isStatic(eachField.getModifiers())) {
462                        if (isStaticFields != null && !isStaticFields.booleanValue()) {
463                                throw new PicoCompositionException("Please make SpecificFieldInjector inject either all non static fields or all non static fields");
464                        }
465                        isStaticFields = Boolean.TRUE;
466                } else {
467                        if (isStaticFields != null && isStaticFields.booleanValue()) {
468                                throw new PicoCompositionException("Please make SpecificFieldInjector inject either all non static fields or all non static fields");
469                        }
470                        isStaticFields = Boolean.FALSE;
471                }
472        }
473                return isStaticFields != null ? isStaticFields : Boolean.FALSE;
474        }

476    /**
477     * Abstract utility class to detect recursion cycles.
478     * Derive from this class and implement {@link ThreadLocalCyclicDependencyGuard#run}.
479     * The method will be called by  {@link ThreadLocalCyclicDependencyGuard#observe}. Select
480     * an appropriate guard for your scope. Any {@link ObjectReference} can be
481     * used as long as it is initialized with  <code>Boolean.FALSE</code>.
482     *
483     * @author J&ouml;rg Schaible
484     */
485    static abstract class ThreadLocalCyclicDependencyGuard<T> extends ThreadLocal<Boolean> {
486
487        protected PicoContainer guardedContainer;
488
489        @Override
490                protected Boolean initialValue() {
491            return Boolean.FALSE;
492        }
493
494        /**
495         * Derive from this class and implement this function with the functionality
496         * to observe for a dependency cycle.
497         *
498         * @return a value, if the functionality result in an expression,
499         *      otherwise just return <code>null</code>
500         */
501        public abstract T run(Object instance);
502
503        /**
504         * Call the observing function. The provided guard will hold the {@link Boolean} value.
505         * If the guard is already <code>Boolean.TRUE</code> a {@link CyclicDependencyException}
506         * will be  thrown.
507         *
508         * @param stackFrame the current stack frame
509         * @return the result of the <code>run</code> method
510         */
511        public final T observe(final Class<?> stackFrame, final Object instance) {
512            if (Boolean.TRUE.equals(get())) {
513                throw new CyclicDependencyException(stackFrame);
514            }
515            T result = null;
516            try {
517                set(Boolean.TRUE);
518                result = run(instance);
519            } catch (final CyclicDependencyException e) {
520                e.push(stackFrame);
521                throw e;
522            } finally {
523                set(Boolean.FALSE);
524            }
525            return result;
526        }
527
528        public void setGuardedContainer(final PicoContainer container) {
529            this.guardedContainer = container;
530        }

534        public static class CyclicDependencyException extends PicoCompositionException {
535        private final List<Class> stack;
536
537        /**
538         * @param element
539         */
540        public CyclicDependencyException(final Class<?> element) {
541            super((Throwable)null);
542            this.stack = new LinkedList<Class>();
543            push(element);
544        }
545
546        /**
547         * @param element
548         */
549        public void push(final Class<?> element) {
550            stack.add(element);
551        }
552
553        public Class[] getDependencies() {
554            return stack.toArray(new Class[stack.size()]);
555        }
556
557        @Override
558                public String getMessage() {
559            return "Cyclic dependency: " + stack.toString();
560        }

563    /**
564     * Exception that is thrown as part of the introspection. Raised if a PicoContainer cannot resolve a
565     * type dependency because the registered {@link com.picocontainer.ComponentAdapter}s are not
566     * distinct.
567     *
568     * @author Paul Hammant
569     * @author Aslak Helles&oslash;y
570     * @author Jon Tirs&eacute;n
571     */
572    public static final class AmbiguousComponentResolutionException extends PicoCompositionException {
573
574
575                private Class<?> component;
576        private final Generic<?> ambiguousDependency;
577        private final Object[] ambiguousComponentKeys;
578                private AccessibleObject accessibleObject;
579
580                /**
581                 * Zero-based parameter #
582                 */
583                private int parameterNumber = -1;
584
585
586
587        /**
588         * Construct a new exception with the ambigous class type and the ambiguous component keys.
589         *
590         * @param ambiguousDependency the unresolved dependency type
591         * @param keys the ambiguous keys.
592         */
593        public AmbiguousComponentResolutionException(final Generic<?> ambiguousDependency, final Object[] keys) {
594            super("");
595            this.ambiguousDependency = ambiguousDependency;
596            this.ambiguousComponentKeys = new Object[keys.length];
597            System.arraycopy(keys, 0, ambiguousComponentKeys, 0, keys.length);
598        }
599
600        /**
601         * @return Returns a string containing the unresolved class type and the ambiguous keys.
602         */
603        @Override
604                public String getMessage() {
605            StringBuffer msg = new StringBuffer();
606            msg.append(component != null ? component : "<not-specified>");
607            msg.append(" needs a '");
608            msg.append(ambiguousDependency.toString());
609            msg.append("' injected");
610            if (parameterNumber > -1) {
611                msg.append(" into parameter #" + parameterNumber);
612                msg.append(" (zero based index) of");
613            }
614
615            if (parameterNumber == -1 && accessibleObject != null) {
616                msg.append(" into");
617            }
618
619            if (accessibleObject != null) {
620                if (accessibleObject instanceof Field) {
621                        msg.append(" field '");
622                } else if (accessibleObject instanceof Constructor) {
623                        msg.append(" constructor '");
624                } else if (accessibleObject instanceof Method) {
625                        msg.append(" method '");
626                } else {
627                        msg.append(" '");
628                }
629                msg.append(accessibleObject);
630            } else {
631                msg.append(" through : <unknown>");
632            }
633            //msg.append(accessibleObject != null ? accessibleObject : "<unknown>");
634            msg.append("', but there are too many choices to inject. These:");
635            msg.append(Arrays.asList(getAmbiguousComponentKeys()));
636            msg.append(", refer http://picocontainer.org/ambiguous-injectable-help.html");
637            return msg.toString();
638        }
639
640        /**
641         * @return Returns the ambiguous component keys as array.
642         */
643        public Object[] getAmbiguousComponentKeys() {
644            return ambiguousComponentKeys;
645        }
