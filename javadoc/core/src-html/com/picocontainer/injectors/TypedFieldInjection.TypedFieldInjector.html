012import static com.picocontainer.Characteristics.immutable;
013
014import java.lang.annotation.Annotation;
015import java.lang.reflect.AccessibleObject;
016import java.lang.reflect.Field;
017import java.lang.reflect.InvocationTargetException;
018import java.lang.reflect.Type;
019import java.security.AccessController;
020import java.security.PrivilegedAction;
021import java.util.ArrayList;
022import java.util.Arrays;
023import java.util.Collections;
024import java.util.List;
025import java.util.Properties;
026
027import com.picocontainer.Characteristics;
028import com.picocontainer.ComponentAdapter;
029import com.picocontainer.ComponentMonitor;
030import com.picocontainer.LifecycleStrategy;
031import com.picocontainer.NameBinding;
032import com.picocontainer.PicoCompositionException;
033import com.picocontainer.annotations.Bind;
034import com.picocontainer.behaviors.AbstractBehavior;
035import com.picocontainer.parameters.ConstructorParameters;
036import com.picocontainer.parameters.FieldParameters;
037import com.picocontainer.parameters.MethodParameters;

039/**
040 * A {@link com.picocontainer.InjectionType} for named fields.
041 *
042 * Use like so: pico.as(injectionFieldNames("field1", "field2")).addComponent(...)
043 *
044 * The factory creates {@link TypedFieldInjector}.
045 *
046 * @author Paul Hammant
047 */
048@SuppressWarnings("serial")
049public class TypedFieldInjection extends AbstractInjectionType {
050
051    private static final String INJECTION_FIELD_TYPES = "injectionFieldTypes";
052
053        public <T> ComponentAdapter<T> createComponentAdapter(final ComponentMonitor monitor,
054                                                   final LifecycleStrategy lifecycle,
055                                                   final Properties componentProps,
056                                                   final Object key,
057                                                   final Class<T> impl,
058                                                   final ConstructorParameters constructorParams, final FieldParameters[] fieldParams, final MethodParameters[] methodParams) throws PicoCompositionException {
059        boolean requireConsumptionOfAllParameters = !(AbstractBehavior.arePropertiesPresent(componentProps, Characteristics.ALLOW_UNUSED_PARAMETERS, false));
060        String fieldTypes = (String) componentProps.remove(INJECTION_FIELD_TYPES);
061        if (fieldTypes == null) {
062            fieldTypes = "";
063        }
064        return wrapLifeCycle(monitor.newInjector(new TypedFieldInjector<T>(key, impl, monitor, fieldTypes, requireConsumptionOfAllParameters, fieldParams
065        )), lifecycle);
066    }
067
068    public static Properties injectionFieldTypes(final String... fieldTypes) {
069        StringBuilder sb = new StringBuilder();
070        for (String fieldType : fieldTypes) {
071            sb.append(" ").append(fieldType);
072        }
073        return immutable(INJECTION_FIELD_TYPES, sb.toString().trim());
074    }
075
076    /**
077     * Injection happens after instantiation, and fields are marked as
078     * injection points via a field type.
079     */
080    public static class TypedFieldInjector<T> extends AbstractFieldInjector<T> {
081
082        private final List<String> classes;
083
084        public TypedFieldInjector(final Object key,
085                                  final Class<?> impl,
086                                  final ComponentMonitor monitor,
087                                  final String classNames,
088                                  final boolean requireUseOfallParameters,
089                                  final FieldParameters... parameters) {
090            super(key, impl, monitor, true, requireUseOfallParameters, parameters);
091            this.classes = Arrays.asList(classNames.trim().split(" "));
092        }
093
094        @Override
095        protected void initializeInjectionMembersAndTypeLists() {
096            injectionMembers = new ArrayList<AccessibleObject>();
097            List<Annotation> bindingIds = new ArrayList<Annotation>();
098            final List<Type> typeList = new ArrayList<Type>();
099            final Field[] fields = getFields();
100            for (final Field field : fields) {
101                if (isTypedForInjection(field)) {
102                    injectionMembers.add(field);
103                    typeList.add(box(field.getType()));
104                    bindingIds.add(getBinding(field));
105                }
106            }
107            injectionTypes = typeList.toArray(new Type[0]);
108            bindings = bindingIds.toArray(new Annotation[0]);
109        }
110
111        private Annotation getBinding(final Field field) {
112            Annotation[] annotations = field.getAnnotations();
113            for (Annotation annotation : annotations) {
114                if (annotation.annotationType().isAnnotationPresent(Bind.class)) {
115                    return annotation;
116                }
117            }
118            return null;
119        }
120
121        protected boolean isTypedForInjection(final Field field) {
122            return classes.contains(field.getType().getName());
123        }
124
125        private Field[] getFields() {
126            return AccessController.doPrivileged(new PrivilegedAction<Field[]>() {
127                public Field[] run() {
128                    return getComponentImplementation().getDeclaredFields();
129                }
130            });
131        }
132
133
134        @Override
135                protected Object injectIntoMember(final AccessibleObject member, final Object componentInstance, final Object toInject)
136            throws IllegalAccessException, InvocationTargetException {
137            Field field = (Field) member;
138            field.setAccessible(true);
139            field.set(componentInstance, toInject);
140            return null;
141        }
142
143        @Override
144        public String getDescriptor() {
145            return "TypedFieldInjector-";
146        }
147
148        @Override
149        protected NameBinding makeParameterNameImpl(final AccessibleObject member) {
150            return new NameBinding() {
151                public String getName() {
152                    return ((Field) member).getName();
153                }
154            };
155        }
156
157        @Override
158                protected Object memberInvocationReturn(final Object lastReturn, final AccessibleObject member, final Object instance) {
159            return instance;
160        }
161
162        List<String> getInjectionFieldTypes() {
163            return Collections.unmodifiableList(classes);
164        }
165
166
167    }
168}
