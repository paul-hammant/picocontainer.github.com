040/**
041 * Creates injector instances, depending on the injection characteristics of the component class.
042 * It will attempt to create a component adapter with - in order of priority:
043 * <ol>
044 *  <li>Annotated field injection: if annotation {@link com.picocontainer.annotations.Inject} is found for field</li>
045 *  <li>Annotated method injection: if annotation {@link com.picocontainer.annotations.Inject} is found for method</li>
046 *  <li>Setter injection: if {@link Characteristics.SDI} is found</li>
047 *  <li>Method injection: if {@link Characteristics.METHOD_INJECTION} if found</li>
048 *  <li>Constructor injection (the default, must find {@link Characteristics.CDI})</li>
049 * </ol>
050 *
051 * @author Paul Hammant
052 * @author Mauro Talevi
053 * @see AnnotatedFieldInjection
054 * @see AnnotatedMethodInjection
055 * @see SetterInjection
056 * @see MethodInjection
057 * @see ConstructorInjection
058 */
059@SuppressWarnings("serial")
060public class AdaptingInjection extends AbstractInjectionType {
061
062        private final ConstructorInjection constructorInjection;
063
064        private final MethodInjection methodInjection;
065
066        private final SetterInjection setterInjection;
067
068        private final AnnotatedMethodInjection annotatedMethodInjection;
069
070        private final AnnotatedFieldInjection annotatedFieldInjection;
071
072        public AdaptingInjection() {
073                constructorInjection = new Jsr330ConstructorInjection();
074                methodInjection = new MethodInjection();
075                setterInjection = new SetterInjection();
076                annotatedMethodInjection = new AnnotatedMethodInjection();
077                annotatedFieldInjection = new AnnotatedFieldInjection();
078        }

081
082        public <T> ComponentAdapter<T> createComponentAdapter(final ComponentMonitor monitor, final LifecycleStrategy lifecycle,
083                                                   final Properties componentProps, final Object key, final Class<T> impl,
084                                                   final ConstructorParameters constructorParams, final FieldParameters[] fieldParams, final MethodParameters[] methodParams) throws PicoCompositionException {
085
086
087                verifyNamedParameters(impl, fieldParams, methodParams);
088
089                ArrayList<InjectionType> injectors = new ArrayList<InjectionType>();
090
091        InjectionType componentAdapter = null;
092
093        componentProps.putAll(Characteristics.ALLOW_UNUSED_PARAMETERS);
094
095        //Pico Style Injections
096        //Must specifically set Characteristics.SDI
097        componentAdapter = setterInjectionAdapter(componentProps);
098        if (componentAdapter != null) {
099                injectors.add(componentAdapter);
100                componentAdapter = null;
101        }
102
103        //Must specifically set Characteristics.METHOD_INJECTION
104        componentAdapter = methodInjectionAdapter(componentProps);
105        if (componentAdapter != null) {
106                injectors.add(componentAdapter);
107                componentAdapter = null;
108        }
109
110
111
112        //JSR 330 injection
113        //Turned on by default
114        componentAdapter = methodAnnotatedInjectionAdapter(impl);
115        if (componentAdapter != null) {
116                injectors.add(componentAdapter);
117                componentAdapter = null;
118        }
119
120
121        //JSR 330 injection
122        //Turned on by default
123        componentAdapter = fieldAnnotatedInjectionAdapter(impl);
124        if (componentAdapter != null) {
125                injectors.add(componentAdapter);
126                componentAdapter = null;
127        }
128
129        injectors.add(defaultInjectionAdapter(componentProps));
130
131
132
133        Collections.reverse(injectors);
134        //return defaultInjectionAdapter(componentProps, monitor, lifecycle, key, impl, parameters);
135        InjectionType[] aArray = injectors.toArray(new InjectionType[injectors.size()]);
136
137
138        //Wrap the static injection behavior.
139        ComponentFactory caf = new CompositeInjection(aArray);
140
141        ComponentAdapter<T> result =  caf
142                .createComponentAdapter(monitor, lifecycle,
143                                componentProps, key, impl,
144                                constructorParams, fieldParams, methodParams);
145
146
147        AbstractBehavior.removePropertiesIfPresent(componentProps, Characteristics.ALLOW_UNUSED_PARAMETERS);
148        return result;
149    }

152        /**
153         * Quick
154         * @throws PicoCompositionException if there's a
155         * @param parameters
156         * @todo This only verifies that the names exist, it doesn't check for name hiding from base class to
157         * sub class.
158         */
159    private void verifyNamedParameters(final Class<?> impl, final FieldParameters[] parameters, final MethodParameters[] methodParameters) {
160        if (parameters == null) {
161                return;
162        }
163        Set<String> allNames = AccessController.doPrivileged(new PrivilegedAction<Set<String>>() {
164
165                        public Set<String> run() {
166                                HashSet<String> result = new HashSet<String>(30);
167                                Class<?> currentImpl = impl;
168                                while (!Object.class.getName().equals(currentImpl.getName())) {
169
170                                        for (Field eachField : currentImpl.getDeclaredFields()) {
171                                                result.add(eachField.getName());
172                                        }
173
174                                        for (Method eachMethod : currentImpl.getDeclaredMethods()) {
175                                                result.add(eachMethod.getName());
176                                        }
177
178                                        currentImpl = currentImpl.getSuperclass();
179                                }
180                                return result;
181                        }});
182
183
184        for (FieldParameters eachParam : parameters) {
185                        if (!allNames.contains(eachParam.getName())) {
186                                throwCompositionException(impl, eachParam);
187                        }
188        }
189
190
191        for (MethodParameters eachParam : methodParameters) {
192                        if (!allNames.contains(eachParam.getName())) {
193                                throwCompositionException(impl, eachParam);
194                        }
195        }
196        }
197
198
199        private void throwCompositionException(final Class<?> impl, final AccessibleObjectParameterSet eachParam) {
200                throw new PicoCompositionException("Cannot locate field or method '"
201                                        + eachParam.getName()
202                                        + "' in type "
203                                        + impl
204                                        + ". \n\tParameter in error: "
205                                        + eachParam);
206        }

211        private  <T> InjectionType defaultInjectionAdapter(final Properties componentProps) {
212        AbstractBehavior.removePropertiesIfPresent(componentProps, Characteristics.CDI);
213        return constructorInjection;
214    }
215
216    private <T> InjectionType setterInjectionAdapter(final Properties componentProps) {
217        if (AbstractBehavior.removePropertiesIfPresent(componentProps, Characteristics.SDI)) {
218                return setterInjection;
219        }
220        return null;
221    }
222
223    private <T> InjectionType methodInjectionAdapter(final Properties componentProps) {
224        if (AbstractBehavior.removePropertiesIfPresent(componentProps, Characteristics.METHOD_INJECTION)) {
225            return methodInjection;
226        }
227        return null;
228    }
229
230
231    private <T> InjectionType methodAnnotatedInjectionAdapter(final Class<T> impl) {
232        if (injectionMethodAnnotated(impl)) {
233                return annotatedMethodInjection;
234        }
235        return null;
236    }
237
238    private <T> InjectionType fieldAnnotatedInjectionAdapter(final Class<T> impl) {
239        if (injectionFieldAnnotated(impl)) {
240                return this.annotatedFieldInjection;
241        }
242
243        return null;
244    }
245
246    private boolean injectionMethodAnnotated(final Class<?> impl) {
247        return  AccessController.doPrivileged(new PrivilegedAction<Boolean>() {
248            @SuppressWarnings("synthetic-access")
249            public Boolean run() {
250                InjectableMethodSelector methodSelector = new InjectableMethodSelector(Inject.class);
251                return (methodSelector.retreiveAllInjectableMethods(impl).size() > 0);
252            }
253        });
254    }
255
256    private boolean injectionFieldAnnotated(final Class<?> impl) {
257        return (Boolean) AccessController.doPrivileged(new PrivilegedAction<Object>() {
258            @SuppressWarnings("synthetic-access")
259            public Object run() {
260                if (impl.isInterface()) {
261                    return false;
262                }
263                Class impl2 = impl;
264                while (impl2 != Object.class) {
265                    if (injectionAnnotated(impl2.getDeclaredFields())) {
266                        return true;
267                    }
268                    impl2 = impl2.getSuperclass();
269                }
270                return false;
271            }
272        });
273    }
274
275    private boolean injectionAnnotated(final AccessibleObject[] objects) {
276        for (AccessibleObject object : objects) {
277            if (object.getAnnotation(Inject.class) != null
278                    || object.getAnnotation(getInjectionAnnotation("javax.inject.Inject")) != null) {
279                return true;
280            }
281        }
282        return false;
283    }
