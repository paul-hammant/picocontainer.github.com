010package com.picocontainer;
011
012
013/**
014 * <p>
015 * Interface responsible for changing monitoring strategy.
016 * It may be implemented by {@link com.picocontainer.PicoContainer containers} and
017 * single {@link com.picocontainer.ComponentAdapter component adapters}.
018 * The choice of supporting the monitor strategy is left to the
019 * implementers of the container and adapters.
020 * </p>
021 *
022 * @author Paul Hammant
023 * @author Joerg Schaible
024 * @author Mauro Talevi
025 */
026public interface ComponentMonitorStrategy {
027
028    /**
029     * Changes the component monitor used
030     * @param monitor the new ComponentMonitor to use
031     * @return the old monitor
032     */
033        ComponentMonitor changeMonitor(ComponentMonitor monitor);
034
035    /**
036     * Returns the monitor currently used
037     * @return The ComponentMonitor currently used
038     */
039    ComponentMonitor currentMonitor();
040
041}
