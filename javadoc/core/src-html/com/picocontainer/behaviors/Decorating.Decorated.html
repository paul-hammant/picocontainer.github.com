011package com.picocontainer.behaviors;
012
013import java.lang.reflect.Type;
014import java.util.Properties;
015
016import com.picocontainer.ComponentAdapter;
017import com.picocontainer.ComponentMonitor;
018import com.picocontainer.Decorator;
019import com.picocontainer.LifecycleStrategy;
020import com.picocontainer.PicoCompositionException;
021import com.picocontainer.PicoContainer;
022import com.picocontainer.parameters.ConstructorParameters;
023import com.picocontainer.parameters.FieldParameters;
024import com.picocontainer.parameters.MethodParameters;
025
026/**
027 * Behavior for Decorating. This factory will create {@link com.picocontainer.behaviors.Decorating.Decorated} that will
028 * allow you to decorate what you like on the component instance that has been created
029 *
030 * @author Paul Hammant
031 */
032public abstract class Decorating extends AbstractBehavior implements Decorator {
033
034    @Override
035        public <T> ComponentAdapter<T> createComponentAdapter(final ComponentMonitor monitor, final LifecycleStrategy lifecycle,
036                                                   final Properties componentProps, final Object key,
037                                                   final Class<T> impl, final ConstructorParameters constructorParams, final FieldParameters[] fieldParams, final MethodParameters[] methodParams) throws PicoCompositionException {
038        return monitor.changedBehavior(new Decorated<T>(super.createComponentAdapter(monitor, lifecycle,
039                componentProps,key, impl, constructorParams, fieldParams, methodParams), this));
040    }
041
042    @SuppressWarnings("serial")
043    public static class Decorated<T> extends AbstractChangedBehavior<T> {
044        private final Decorator decorator;
045
046        public Decorated(final ComponentAdapter<T> delegate, final Decorator decorator) {
047            super(delegate);
048            this.decorator = decorator;
049        }
050
051        @Override
052                public T getComponentInstance(final PicoContainer container, final Type into)
053                throws PicoCompositionException {
054            T instance = super.getComponentInstance(container, into);
055            decorator.decorate(instance);
056            return instance;
057        }
058
059        public String getDescriptor() {
060            return "Decorated";
061        }
062
063    }
064
065}
