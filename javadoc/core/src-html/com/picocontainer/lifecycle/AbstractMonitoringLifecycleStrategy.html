008package com.picocontainer.lifecycle;
009
010import java.io.Serializable;
011
012import com.picocontainer.ComponentAdapter;
013import com.picocontainer.ComponentMonitor;
014import com.picocontainer.ComponentMonitorStrategy;
015import com.picocontainer.LifecycleStrategy;
016
017/**
018 * Abstract base class for lifecycle strategy implementation supporting a {@link ComponentMonitor}.
019 *
020 * @author J&ouml;rg Schaible
021 */
022@SuppressWarnings("serial")
023public abstract class AbstractMonitoringLifecycleStrategy implements LifecycleStrategy, ComponentMonitorStrategy, Serializable {
024
025        /**
026         * Component monitor that receives lifecycle state.
027         */
028    private ComponentMonitor monitor;
029
030    /**
031     * Construct a AbstractMonitoringLifecycleStrategy.
032     *
033     * @param monitor the monitor to use
034     * @throws NullPointerException if the monitor is <code>null</code>
035     */
036    public AbstractMonitoringLifecycleStrategy(final ComponentMonitor monitor) {
037        changeMonitor(monitor);
038    }
039
040    /**
041     * Swaps the current monitor with a replacement.
042     * @param newMonitor The new monitor.
043     * @throws NullPointerException if the passed in monitor is null.
044     */
045    public ComponentMonitor changeMonitor(final ComponentMonitor newMonitor) {
046        if (newMonitor == null) {
047            throw new NullPointerException("Monitor is null");
048        }
049        ComponentMonitor oldValue = monitor;
050        this.monitor = newMonitor;
051        
052        return oldValue;
053    }
054
055    public ComponentMonitor currentMonitor() {
056        return monitor;
057    }
058
059    public boolean isLazy(final ComponentAdapter<?> adapter) {
060        return false;
061    }
062
063}
