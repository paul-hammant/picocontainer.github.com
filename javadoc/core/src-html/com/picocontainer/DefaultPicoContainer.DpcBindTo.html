057/**
058 * <p/>
059 * The Standard {@link PicoContainer}/{@link MutablePicoContainer} implementation.
060 * Constructing a container c with a parent p container will cause c to look up components
061 * in p if they cannot be found inside c itself.
062 * </p>
063 * <p/>
064 * Using {@link Class} objects as keys to the various registerXXX() methods makes
065 * a subtle semantic difference:
066 * </p>
067 * <p/>
068 * If there are more than one registered components of the same type and one of them are
069 * registered with a {@link java.lang.Class} key of the corresponding type, this addComponent
070 * will take precedence over other components during type resolution.
071 * </p>
072 * <p/>
073 * Another place where keys that are classes make a subtle difference is in
074 * {@link com.picocontainer.behaviors.ImplementationHiding.HiddenImplementation}.
075 * </p>
076 * <p/>
077 * This implementation of {@link MutablePicoContainer} also supports
078 * {@link ComponentMonitorStrategy}.
079 * </p>
080 *
081 * @author Paul Hammant
082 * @author Aslak Helles&oslash;y
083 * @author Jon Tirs&eacute;n
084 * @author Thomas Heller
085 * @author Mauro Talevi
086 */

217    public DefaultPicoContainer(final PicoContainer parent, final LifecycleStrategy lifecycle, final ComponentMonitor monitor, final ComponentFactory... componentFactories) {
218        if (componentFactories.length == 0) {
219            throw new NullPointerException("at least one componentFactory");
220        }
221        int i = componentFactories.length -1;
222        ComponentFactory componentFactory = componentFactories[i];
223        while (i > 0) {
224            try {
225                componentFactory = ((Behavior) componentFactories[i-1]).wrap(componentFactory);
226            } catch (ClassCastException e) {
227                throw new PicoCompositionException("Check the order of the BehaviorFactories " +
228                        "in the varargs list of ComponentFactories. Index " + (i-1)
229                        + " ("+componentFactories[i-1].getClass().getName()+") should be a BehaviorFactory but is not.");
230            }
231            i--;
232        }
233        if (componentFactory == null) {
234            throw new NullPointerException("one of the componentFactories");
235        }
236        if (lifecycle == null) {
237            throw new NullPointerException("lifecycle");
238        }
239        this.componentFactory = componentFactory;
240        this.lifecycle = lifecycle;
241        this.parent = parent;
242        if (parent != null && !(parent instanceof EmptyPicoContainer)) {
243            this.parent = new ImmutablePicoContainer(parent);
244        }
245        this.monitor = monitor;
246    }

301    @SuppressWarnings({ "rawtypes", "unchecked" })
302        public final ComponentAdapter<?> getComponentAdapter (Object key) {
303        if (key instanceof Generic) {
304            key = ((Generic) key).getType();
305        }
306        ComponentAdapter<?> adapter = getComponentKeyToAdapterCache().get(key);
307        if (adapter == null && parent != null) {
308            adapter = getParent().getComponentAdapter(key);
309            if (adapter != null) {
310                adapter = new KnowsContainerAdapter(adapter, getParent());
311            }
312        }
313        if (adapter == null) {
314            Object inst = monitor.noComponentFound(this, key);
315            if (inst != null) {
316                adapter = new LateInstance(key, inst);
317            }
318        }
319        return adapter;
320    }

412    /**
413     * {@inheritDoc} *
414     */
415    private <T> ComponentAdapter<T> getComponentAdapter(final Generic<T> componentType, final NameBinding componentNameBinding, final Class<? extends Annotation> binding) {
416        // See http://jira.codehaus.org/secure/ViewIssue.jspa?key=PICO-115
417        ComponentAdapter<T> adapterByKey = (ComponentAdapter<T>) getComponentAdapter(componentType);
418        if (adapterByKey != null) {
419            return adapterByKey;
420        }
421
422        List<ComponentAdapter<T>> found = binding == null ? getComponentAdapters(componentType) : getComponentAdapters(componentType, binding);
423
424        if (found.size() == 1) {
425            return found.get(0);
426        } else if (found.isEmpty()) {
427            if (parent != null) {
428                return getParent().getComponentAdapter(componentType, componentNameBinding);
429            } else {
430                return null;
431            }
432        } else {
433            if (componentNameBinding != null) {
434                String parameterName = componentNameBinding.getName();
435                if (parameterName != null) {
436                    ComponentAdapter<?> ca = getComponentAdapter(parameterName);
437                    if (ca != null && JTypeHelper.isAssignableFrom(componentType, ca.getComponentImplementation())) {
438                        return (ComponentAdapter<T>) ca;
439                    }
440                }
441            }
442            Class<?>[] foundClasses = new Class[found.size()];
443            for (int i = 0; i < foundClasses.length; i++) {
444                foundClasses[i] = found.get(i).getComponentImplementation();
445            }
446
447            throw new AbstractInjector.AmbiguousComponentResolutionException(componentType, foundClasses);
448        }
449    }

483    /**
484     * {@inheritDoc} *
485     */
486    public <T> List<ComponentAdapter<T>> getComponentAdapters(final Generic<T> componentType, final Class<? extends Annotation> binding) {
487        if (componentType == null) {
488            return Collections.emptyList();
489        }
490        List<ComponentAdapter<T>> found = new ArrayList<ComponentAdapter<T>>();
491        for (ComponentAdapter<?> componentAdapter : getComponentAdapters()) {
492            Object key = componentAdapter.getComponentKey();
493
494            //JSR 330 Provider compatibility... we have to be able to return both the providers that provide
495            //the type as well as the actual types themselves.
496            Class<?> implementation = componentAdapter.getComponentImplementation();
497
498
499            boolean compatible = JTypeHelper.isAssignableFrom(componentType, implementation);
500            if (!compatible && componentAdapter.findAdapterOfType(ProviderAdapter.class) != null) {
501                //If provider
502                //Todo: Direct access of provider adapter... work around.
503                ProviderAdapter adapter = componentAdapter.findAdapterOfType(ProviderAdapter.class);
504                compatible = JTypeHelper.isAssignableFrom(componentType, adapter.getProviderReturnType());
505            }
506
507            if (compatible &&
508                    (!(key instanceof Key) || ((((Key<?>) key).getAnnotation() == null || binding == null ||
509                            ((Key<?>) key).getAnnotation() == binding)))) {
510                found.add((ComponentAdapter<T>) componentAdapter);
511            }
512        }
513        return found;
514    }

516    protected MutablePicoContainer addAdapterInternal(final ComponentAdapter<?> componentAdapter) {
517        Object key = componentAdapter.getComponentKey();
518        if (getComponentKeyToAdapterCache().containsKey(key)) {
519            throw new PicoCompositionException("Duplicate Keys not allowed. Duplicate for '" + key + "'");
520        }
521        getModifiableComponentAdapterList().add(componentAdapter);
522        getComponentKeyToAdapterCache().put(key, componentAdapter);
523        return this;
524    }

554        Properties tmpProperties = (Properties) properties.clone();
555        removeGenericPropertiesThatWeDontCareAbout(tmpProperties);
556
557
558        if (AbstractBehavior.removePropertiesIfPresent(tmpProperties, Characteristics.NONE) == false && componentFactory instanceof Behavior) {
559            MutablePicoContainer container = addAdapterInternal(((Behavior) componentFactory).addComponentAdapter(
560                    monitor,
561                    lifecycle,
562                    tmpProperties,
563                    componentAdapter));
564            throwIfPropertiesLeft(tmpProperties);
565            return container;
566        } else {
567            return addAdapterInternal(componentAdapter);
568        }

607        }
608        return addComponent(clazz, implOrInstance, props, null, null, null);

638    private MutablePicoContainer addComponent(Object key,
639                                              final Object implOrInstance,
640                                              final Properties properties,
641                                              final ConstructorParameters constructorParameters,
642                                              final FieldParameters[] fieldParameters,
643                                              final MethodParameters[] methodParameters) {
644
645        Parameter[] tweakedParameters = (constructorParameters != null) ? constructorParameters.getParams() : null;
646
647        if (key instanceof Generic) {
648            key = Generic.get(((Generic) key).getType());
649        }
650        if (tweakedParameters != null && tweakedParameters.length == 0) {
651            tweakedParameters = null; // backwards compatibility!  solve this better later - Paul
652        }
653
654        //New replacement for Parameter.ZERO.
655        if (tweakedParameters != null && tweakedParameters.length == 1 && DefaultConstructorParameter.INSTANCE.equals(tweakedParameters[0])) {
656            tweakedParameters = new Parameter[0];
657        }
658
659
660
661        if (implOrInstance instanceof Class) {
662            Properties tmpProperties = (Properties) properties.clone();
663            ComponentAdapter<?> adapter = componentFactory.createComponentAdapter(monitor,
664                    lifecycle,
665                    tmpProperties,
666                    key,
667                    (Class<?>) implOrInstance,
668                    new ConstructorParameters(tweakedParameters), fieldParameters, methodParameters);
669            removeGenericPropertiesThatWeDontCareAbout(tmpProperties);
670            throwIfPropertiesLeft(tmpProperties);
671            if (lifecycleState.isStarted()) {
672                addAdapterIfStartable(adapter);
673                potentiallyStartAdapter(adapter);
674            }
675            return addAdapterInternal(adapter);
676        } else {
677            ComponentAdapter<?> adapter =
678                    new InstanceAdapter<Object>(key, implOrInstance, lifecycle, monitor);
679            if (lifecycleState.isStarted()) {
680                addAdapterIfStartable(adapter);
681                potentiallyStartAdapter(adapter);
682            }
683            return addAdapter(adapter, properties);
684        }
685    }

687        private void removeGenericPropertiesThatWeDontCareAbout(final Properties tmpProperties) {
688                AbstractBehavior.removePropertiesIfPresent(tmpProperties, Characteristics.USE_NAMES);
689        AbstractBehavior.removePropertiesIfPresent(tmpProperties, Characteristics.STATIC_INJECTION);
690        }

763    public <T> List<T> getComponents(final Class<T> componentType) {
764        if (componentType == null) {
765            return Collections.emptyList();
766        }
767
768        Map<ComponentAdapter<T>, T> adapterToInstanceMap = new HashMap<ComponentAdapter<T>, T>();
769        List<T> result = new ArrayList<T>();
770        synchronized (this) {
771            for (ComponentAdapter<?> componentAdapter : getModifiableComponentAdapterList()) {
772                if (componentType.isAssignableFrom(componentAdapter.getComponentImplementation())) {
773                    ComponentAdapter<T> typedComponentAdapter = (ComponentAdapter<T>) componentAdapter;
774                    T componentInstance = getLocalInstance(typedComponentAdapter);
775                    adapterToInstanceMap.put(typedComponentAdapter, componentInstance);
776                }
777            }
778            for (ComponentAdapter<?> componentAdapter : getOrderedComponentAdapters()) {
779                final T componentInstance = adapterToInstanceMap.get(componentAdapter);
780                if (componentInstance != null) {
781                    // may be null in the case of the "implicit" addAdapter
782                    // representing "this".
783                    result.add(componentInstance);
784                }
785            }
786        }
787        return result;
788    }

869    private Object getInstance(final ComponentAdapter<?> componentAdapter, final Generic<?> key, final Type into) {
870        // check whether this is our adapter
871        // we need to check this to ensure up-down dependencies cannot be followed
872        final boolean isLocal = getModifiableComponentAdapterList().contains(componentAdapter);
873
874        if (isLocal || componentAdapter instanceof LateInstance) {
875            Object instance;
876            try {
877                if (componentAdapter instanceof FactoryInjector) {
878                    instance = ((FactoryInjector) componentAdapter).getComponentInstance(this, into);
879                } else {
880                    instance = componentAdapter.getComponentInstance(this, into);
881                }
882            } catch (AbstractInjector.CyclicDependencyException e) {
883                if (parent != null) {
884                    instance = getParent().getComponentInto(componentAdapter.getComponentKey(), into);
885                    if (instance != null) {
886                        return instance;
887                    }
888                }
889                throw e;
890            }
891            addOrderedComponentAdapter(componentAdapter);
892
893            return instance;
894        } else if (parent != null) {
895                Object componentKey = key;
896                if (componentKey == null) {
897                        componentKey = componentAdapter.getComponentKey();
898                }
899            return getParent().getComponentInto(componentKey, into);
900        }
901
902        return null;
903    }

926     * Start the components of this PicoContainer and all its logical child containers.
927     * The starting of the child container is only attempted if the parent
928     * container start successfully.  The child container for which start is attempted
929     * is tracked so that upon stop, only those need to be stopped.
930     * The lifecycle operation is delegated to the component adapter,
931     * if it is an instance of {@link ChangedBehavior lifecycle manager}.
932     * The actual {@link LifecycleStrategy lifecycle strategy} supported
933     * depends on the concrete implementation of the adapter.
934     *
935     * @see ChangedBehavior
936     * @see LifecycleStrategy
937     * @see #makeChildContainer()
938     * @see #addChildContainer(PicoContainer)
939     * @see #removeChildContainer(PicoContainer)
940     */
941    public synchronized void start() {
942
943        lifecycleState.starting(getName());
944
945        startAdapters();
946        childrenStarted.clear();
947        for (PicoContainer child : children) {
948            childrenStarted.add(new WeakReference<PicoContainer>(child));
949            if (child instanceof Startable) {
950                ((Startable) child).start();
951            }
952        }
953   }

955    /**
956     * Stop the components of this PicoContainer and all its logical child containers.
957     * The stopping of the child containers is only attempted for those that have been
958     * started, possibly not successfully.
959     * The lifecycle operation is delegated to the component adapter,
960     * if it is an instance of {@link ChangedBehavior lifecycle manager}.
961     * The actual {@link LifecycleStrategy lifecycle strategy} supported
962     * depends on the concrete implementation of the adapter.
963     *
964     * @see ChangedBehavior
965     * @see LifecycleStrategy
966     * @see #makeChildContainer()
967     * @see #addChildContainer(PicoContainer)
968     * @see #removeChildContainer(PicoContainer)
969     */
970    public synchronized void stop() {
971        lifecycleState.stopping(getName());
972        try {
973                for (PicoContainer child : children) {
974                    if (childStarted(child)) {
975                        if (child instanceof Startable) {
976                            ((Startable) child).stop();
977                        }
978                    }
979                }
980                
981        } finally {
982                try {
983                        stopAdapters();
984                } finally {
985                        lifecycleState.stopped();
986                }
987        }
988    }

990    /**
991     * Checks the status of the child container to see if it's been started
992     * to prevent IllegalStateException upon stop
993     *
994     * @param child the child PicoContainer
995     * @return A boolean, <code>true</code> if the container is started
996     */
997    private boolean childStarted(final PicoContainer child) {
998        for (WeakReference<PicoContainer> eachChild : childrenStarted) {
999            PicoContainer ref = eachChild.get();
1000            if (ref == null) {
1001                continue;
1002            }
1003
1004            if (child.equals(ref)) {
1005                return true;
1006            }
1007        }
1008        return false;
1009    }

1012     * Dispose the components of this PicoContainer and all its logical child containers.
1013     * The lifecycle operation is delegated to the component adapter,
1014     * if it is an instance of {@link ChangedBehavior lifecycle manager}.
1015     * The actual {@link LifecycleStrategy lifecycle strategy} supported
1016     * depends on the concrete implementation of the adapter.
1017     *
1018     * @see ChangedBehavior
1019     * @see LifecycleStrategy
1020     * @see #makeChildContainer()
1021     * @see #addChildContainer(PicoContainer)
1022     * @see #removeChildContainer(PicoContainer)
1023     */
1024    public synchronized void dispose() {
1025        if (lifecycleState.isStarted()) {
1026            stop();
1027        }
1028
1029        lifecycleState.disposing(getName());
1030
1031        try {
1032                for (PicoContainer child : children) {
1033                    if (child instanceof MutablePicoContainer) {
1034                        ((Disposable) child).dispose();
1035                    }
1036                }
1037        } finally {
1038                try {
1039                        disposeAdapters();
1040                componentFactory.dispose();
1041                } finally {
1042                lifecycleState.disposed();
1043                }
1044
1045        }
1046
1047        
1048    }

1066    /**
1067     * Checks for identical references in the child container.  It doesn't
1068     * traverse an entire hierarchy, namely it simply checks for child containers
1069     * that are equal to the current container.
1070     *
1071     * @param child
1072     */
1073    private void checkCircularChildDependencies(final PicoContainer child) {
1074        final String MESSAGE = "Cannot have circular dependency between parent %s and child: %s";
1075        if (child == this) {
1076            throw new IllegalArgumentException(String.format(MESSAGE, this, child));
1077        }
1078
1079        //Todo: Circular Import Dependency on AbstractDelegatingPicoContainer
1080        if (child instanceof AbstractDelegatingPicoContainer) {
1081            AbstractDelegatingPicoContainer delegateChild = (AbstractDelegatingPicoContainer) child;
1082            while (delegateChild != null) {
1083                PicoContainer delegateInstance = delegateChild.getDelegate();
1084                if (this == delegateInstance) {
1085                    throw new IllegalArgumentException(String.format(MESSAGE, this, child));
1086                }
1087                if (delegateInstance instanceof AbstractDelegatingPicoContainer) {
1088                    delegateChild = (AbstractDelegatingPicoContainer) delegateInstance;
1089                } else {
1090                    delegateChild = null;
1091                }
1092            }
1093        }
1094
1095    }

1108    public boolean removeChildContainer(final PicoContainer child) {
1109        final boolean result = children.remove(child);
1110        WeakReference<PicoContainer> foundRef = null;
1111        for (WeakReference<PicoContainer> eachChild : childrenStarted) {
1112            PicoContainer ref = eachChild.get();
1113            if (ref.equals(child)) {
1114                foundRef = eachChild;
1115                break;
1116            }
1117        }
1118
1119        if (foundRef != null) {
1120            childrenStarted.remove(foundRef);
1121        }
1122
1123        return result;
1124    }

1142
1143        //TODO Pico 3 : change accept signatures to allow abort at any point in the traversal.
1144        boolean shouldContinue = visitor.visitContainer(this);
1145        if (!shouldContinue) {
1146            return;
1147        }
1148
1149
1150        componentFactory.accept(visitor); // will cascade through behaviors
1151
1152
1153        final List<ComponentAdapter<?>> componentAdapters = new ArrayList<ComponentAdapter<?>>(getComponentAdapters());
1154        for (ComponentAdapter<?> componentAdapter : componentAdapters) {
1155            componentAdapter.accept(visitor);
1156        }
1157        final List<PicoContainer> allChildren = new ArrayList<PicoContainer>(children);
1158        for (PicoContainer child : allChildren) {
1159            child.accept(visitor);
1160        }
1161    }

1199    /**
1200     * Loops over all component adapters and invokes
1201     * start(PicoContainer) method on the ones which are LifecycleManagers
1202     */
1203    private void startAdapters() {
1204        Collection<ComponentAdapter<?>> adapters = getComponentAdapters();
1205        for (ComponentAdapter<?> adapter : adapters) {
1206            addAdapterIfStartable(adapter);
1207        }
1208        adapters = getOrderedComponentAdapters();
1209        // clone the adapters
1210        List<ComponentAdapter<?>> adaptersClone = new ArrayList<ComponentAdapter<?>>(adapters);
1211        for (final ComponentAdapter<?> adapter : adaptersClone) {
1212            potentiallyStartAdapter(adapter);
1213        }
1214    }

1243     * stop(PicoContainer) method on the ones which are LifecycleManagers
1244     */
1245    private void stopAdapters() {
1246        for (int i = getOrderedComponentAdapters().size() - 1; 0 <= i; i--) {
1247            ComponentAdapter<?> adapter = getOrderedComponentAdapters().get(i);
1248            if (adapter instanceof ComponentLifecycle) {
1249                ComponentLifecycle<?> componentLifecycle = (ComponentLifecycle<?>) adapter;
1250                if (componentLifecycle.componentHasLifecycle() && componentLifecycle.isStarted()) {
1251                    componentLifecycle.stop(DefaultPicoContainer.this);
1252                }
1253            }
1254        }
1255    }
