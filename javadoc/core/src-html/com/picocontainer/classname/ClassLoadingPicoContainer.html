008package com.picocontainer.classname;
009
010import java.net.URL;
011
012import com.picocontainer.MutablePicoContainer;
013import com.picocontainer.PicoContainer;
014
015/**
016 * A ClassLoadingPicoContainer extends PicoContainer with classloader juggling capability
017 *
018 * @author Paul Hammant
019 * @author Aslak Helles&oslash;y
020 */
021public interface ClassLoadingPicoContainer extends MutablePicoContainer {
022
023    /**
024     * Adds a new URL that will be used in classloading
025     *
026     * @param url url of the jar to find components in.
027     * @return ClassPathElement to add permissions to (subject to security
028     *         policy)
029     */
030    ClassPathElement addClassLoaderURL(URL url);
031
032    /**
033     * Returns class loader that is the aggregate of the URLs added.
034     *
035     * @return A ClassLoader
036     */
037    ClassLoader getComponentClassLoader();
038
039    /**
040     * Make a child container with a given name
041     *
042     * @param name the container name
043     * @return The ScriptedPicoContainer
044     */
045    ClassLoadingPicoContainer makeChildContainer(String name);
046
047    /**
048     * Addes a child container with a given name
049     *
050     * @param name the container name
051     * @param child the child PicoContainer
052     */
053    ClassLoadingPicoContainer addChildContainer(String name, PicoContainer child);
054
055}
