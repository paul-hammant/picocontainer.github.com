008package com.picocontainer.classname;
009
010/**
011 * ClassName is a simple wrapper for a class name which is used as a key in
012 * the registration of components in PicoContainer.
013 *
014 * @author Paul Hammant
015 */
016public class ClassName implements CharSequence {
017    private final String className;
018
019    public ClassName(final String className) {
020        this.className = className;
021    }
022
023    public int length() {
024        return className.length();
025    }
026
027    public char charAt(final int ix) {
028        return className.charAt(ix);
029    }
030
031    public CharSequence subSequence(final int from, final int to) {
032        return className.subSequence(from, to);
033    }
034
035    @Override
036        public String toString() {
037        return className.toString();
038    }
039
040    @Override
041        public int hashCode() {
042        return className.hashCode();
043    }
044
045    @Override
046        public boolean equals(final Object o) {
047        return className.equals(o);
048    }
049}
