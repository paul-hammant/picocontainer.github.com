009package com.picocontainer.gems.constraints;
010
011
012import com.googlecode.jtype.Generic;
013import com.picocontainer.ComponentAdapter;
014import com.picocontainer.PicoVisitor;
015import com.picocontainer.parameters.CollectionComponentParameter;
016
017/**
018 * Constraint that collects/aggregates dependencies to as many components
019 * that satisfy the given constraint.
020 *
021 * @author Nick Sieger
022 * @author J&ouml;rg Schaible
023 */
024@SuppressWarnings("serial")
025public final class CollectionConstraint extends CollectionComponentParameter implements Constraint {
026
027        protected final Constraint constraint;
028
029    public CollectionConstraint(final Constraint constraint) {
030        this(constraint, false);
031    }
032
033    public CollectionConstraint(final Constraint constraint, final boolean emptyCollection) {
034        super(Generic.get(Object.class), emptyCollection);
035        this.constraint = constraint;
036    }
037
038    @Override
039        public boolean evaluate(final ComponentAdapter adapter) {
040        return constraint.evaluate(adapter);
041    }
042
043    @Override
044        public void accept(final PicoVisitor visitor) {
045        super.accept(visitor);
046        constraint.accept(visitor);
047    }
048}
