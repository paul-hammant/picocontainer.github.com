001package com.picocontainer.gems;
002
003import static com.picocontainer.Characteristics.FALSE;
004import static com.picocontainer.Characteristics.TRUE;
005import static com.picocontainer.Characteristics.immutable;
006
007import java.util.Properties;
008
009/**
010 * A list of properties to allow switching on and off different characteristics at container
011 * construction time.
012 * @author Michael Rimov
013 */
014public final class GemsCharacteristics {
015
016        private static String _JMX = "jmx";
017
018    private static final String _HOT_SWAP = "hotswap";
019
020
021    private static final String _POOL = "pooled";
022
023    /**
024     * Turn off behavior for {@link com.picocontainer.gems.jmx.JMXExposing JMXExposing}
025     */
026    public static final Properties NO_JMX = immutable(_JMX, FALSE);
027
028    /**
029     * Turn on behavior for {@link com.picocontainer.gems.jmx.JMXExposing JMXExposing}
030     */
031    public static final Properties JMX = immutable(_JMX, TRUE);
032
033
034    /**
035     * Turn on hot-swapping behavior.
036     */
037    public static final Properties HOT_SWAP = immutable(_HOT_SWAP, TRUE);
038
039    /**
040     * Turn off hot swapping behavior.
041     */
042    public static final Properties NO_HOT_SWAP = immutable(_HOT_SWAP, FALSE);
043
044    /**
045     * Turn on pooling behavior.
046     */
047    public static final Properties POOL = immutable(_POOL, TRUE);
048
049    /**
050     * Turn off pooling behavior.
051     */
052    public static final Properties NO_POOL = immutable(_POOL, FALSE);
053
054}
