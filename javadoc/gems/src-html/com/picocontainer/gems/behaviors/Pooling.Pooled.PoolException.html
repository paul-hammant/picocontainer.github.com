055    @Override
056        public <T> ComponentAdapter<T> createComponentAdapter(final ComponentMonitor monitor, final LifecycleStrategy lifecycle, final Properties componentProps, final Object key, final Class<T> impl, final ConstructorParameters constructorParams, final FieldParameters[] fieldParams, final MethodParameters[] methodParams)
057            throws PicoCompositionException {
058        ComponentAdapter<T> delegate = super.createComponentAdapter(monitor, lifecycle, componentProps, key, impl, constructorParams, fieldParams, methodParams);
059
060        if (AbstractBehavior.removePropertiesIfPresent(componentProps, GemsCharacteristics.NO_POOL)) {
061                return delegate;
062                }
063
064        AbstractBehavior.removePropertiesIfPresent(componentProps, GemsCharacteristics.POOL);
065        Pooled<T> behavior = new Pooled<T>(delegate, poolContext);
066        //TODO
067        //Characteristics.HIDE.setProcessedIn(componentCharacteristics);
068        return monitor.changedBehavior(behavior);
069    }
070
071    @Override
072        public <T> ComponentAdapter<T> addComponentAdapter(final ComponentMonitor monitor, final LifecycleStrategy lifecycle,
073                                                final Properties componentProps, final ComponentAdapter<T> adapter) {
074
075        if (AbstractBehavior.removePropertiesIfPresent(componentProps, GemsCharacteristics.NO_POOL)) {
076                return super.addComponentAdapter(monitor, lifecycle, componentProps, adapter);
077                }
078
079        AbstractBehavior.removePropertiesIfPresent(componentProps, GemsCharacteristics.POOL);
080        return monitor.changedBehavior(new Pooled<T>(super.addComponentAdapter(monitor, lifecycle, componentProps, adapter), poolContext));
081    }

083    /**
084     * {@link com.picocontainer.ComponentAdapter} implementation that pools components.
085     * <p>
086     * The implementation utilizes a delegated ComponentAdapter to create the instances of the pool. The
087     * pool can be configured to grow unlimited or to a maximum size. If a component is requested from
088     * this adapter, the implementation returns an available instance from the pool or will create a
089     * new one, if the maximum pool size is not reached yet. If none is available, the implementation
090     * can wait a defined time for a returned object before it throws a {@link com.picocontainer.gems.behaviors.Pooling.Pooled.PoolException}.
091     * </p>
092     * <p>
093     * This implementation uses the {@link com.thoughtworks.proxy.toys.pool.Pool} toy from the <a
094     * href="http://proxytoys.codehaus.org">ProxyToys</a> project. This ensures, that any component,
095     * that is out of scope will be automatically returned to the pool by the garbage collector.
096     * Additionally will every component instance also implement
097     * {@link com.thoughtworks.proxy.toys.pool.Poolable}, that can be used to return the instance
098     * manually. After returning an instance it should not be used in client code anymore.
099     * </p>
100     * <p>
101     * Before a returning object is added to the available instances of the pool again, it should be
102     * reinitialized to a normalized state. By providing a proper Resetter implementation this can be
103     * done automatically. If the object cannot be reused anymore it can also be dropped and the pool
104     * may request a new instance.
105     * </p>
106     * <p>
107     * The pool supports components with a lifecycle. If the delegated {@link com.picocontainer.ComponentAdapter}
108     * implements a {@link com.picocontainer.LifecycleStrategy}, any component retrieved form the pool will be started
109     * before and stopped again, when it returns back into the pool. Also if a component cannot be
110     * resetted it will automatically be disposed. If the container of the pool is disposed, that any
111     * returning object is also disposed and will not return to the pool anymore. Note, that current
112     * implementation cannot dispose pooled objects.
113     * </p>
114     *
115     * @author J&ouml;rg Schaible
116     * @author Aslak Helles&oslash;y
117     */
118    public static final class Pooled<T> extends AbstractChangedBehavior<T> {

262        /**
263         * Construct a Pooled. Remember, that the implementation will request new
264         * components from the delegate as long as no component instance is available in the pool and
265         * the maximum pool size is not reached. Therefore the delegate may not return the same
266         * component instance twice. Ensure, that the used {@link com.picocontainer.ComponentAdapter} does not cache.
267         *
268         * @param delegate the delegated ComponentAdapter
269         * @param context the {@link com.picocontainer.gems.behaviors.Pooling.Pooled.Context} of the pool
270         * @throws IllegalArgumentException if the maximum pool size or the serialization mode is
271         *             invalid
272         */
273        public Pooled(final ComponentAdapter<T> delegate, final Context context) {
274            super(delegate);
275            this.maxPoolSize = context.getMaxSize();
276            this.waitMilliSeconds = context.getMaxWaitInMilliseconds();
277            this.autostartGC = context.autostartGC();
278            this.serializationMode = context.getSerializationMode();
279            if (maxPoolSize <= 0) {
280                throw new IllegalArgumentException("Invalid maximum pool size");
281            }
282            started = false;
283            disposed = false;
284            delegateHasLifecylce = delegate instanceof LifecycleStrategy
285                    && ((LifecycleStrategy)delegate)
286                            .hasLifecycle(delegate.getComponentImplementation());
287            components = new ArrayList<Object>();
288
289            final Class type = delegate.getComponentKey() instanceof Class ? (Class)delegate
290                    .getComponentKey() : delegate.getComponentImplementation();
291            this.pool = Pool.create(type).resettedBy(makeResetted(context)).mode(serializationMode).build(context.getProxyFactory());
292
293        }
294
295        private Resetter<T> makeResetted(final Context context) {
296            final Resetter<T> resetter = context.getResetter();
297            return delegateHasLifecylce ? new LifecycleResetter<T>(this, resetter) : resetter;
298        }

301         * Construct an empty ComponentAdapter, used for serialization with reflection only.

309        /**
310         * {@inheritDoc}
311         * <p>
312         * As long as the maximum size of the pool is not reached and the pool is exhausted, the
313         * implementation will request its delegate for a new instance, that will be managed by the
314         * pool. Only if the maximum size of the pool is reached, the implementation may wait (depends
315         * on the initializing {@link com.picocontainer.gems.behaviors.Pooling.Pooled.Context}) for a returning object.
316         * </p>
317         *
318         * @throws com.picocontainer.gems.behaviors.Pooling.Pooled.PoolException if the pool is exhausted or waiting for a returning object timed out or
319         *             was interrupted
320         */
321        @Override
322        public T getComponentInstance(final PicoContainer container, final Type into) {
323            if (delegateHasLifecylce) {
324                if (disposed) {
325                                        throw new IllegalStateException("Already disposed");
326                                }
327            }
328            T componentInstance;
329            long now = System.currentTimeMillis();
330            boolean gc = autostartGC;
331            while (true) {
332                synchronized (pool) {
333                    componentInstance = (T) pool.get();
334                    if (componentInstance != null) {
335                        break;
336                    }
337                    if (maxPoolSize > pool.size()) {
338                        final Object component = super.getComponentInstance(container, into);
339                        if (delegateHasLifecylce) {
340                            components.add(component);
341                            if (started) {
342                                start(component);
343                            }
344                        }
345                        pool.add(component);
346                    } else if (!gc) {
347                        long after = System.currentTimeMillis();
348                        if (waitMilliSeconds < 0) {
349                            throw new PoolException("Pool exhausted");
350                        }
351                        if (waitMilliSeconds > 0 && after - now > waitMilliSeconds) {
352                            throw new PoolException("Time out wating for returning object into pool");
353                        }
354                        try {
355                            pool.wait(waitMilliSeconds); // Note, the pool notifies after an object
356                                                            // was returned
357                        } catch (InterruptedException e) {
358                            // give the client code of the current thread a chance to abort also
359                            Thread.currentThread().interrupt();
360                            throw new PoolException(
361                                    "Interrupted waiting for returning object into the pool", e);
362                        }
363                    } else {
364                        System.gc();
365                        gc = false;
366                    }
367                }
368            }
369            return componentInstance;
370        }

386        static final class LifecycleResetter<T> implements Resetter<T>, Serializable {
387            private final Resetter<T> delegate;
388            private final Pooled adapter;
389
390            LifecycleResetter(final Pooled adapter, final Resetter<T> delegate) {
391                this.adapter = adapter;
392                this.delegate = delegate;
393            }
394
395            public boolean reset(final T object) {
396                final boolean result = delegate.reset(object);
397                if (!result || adapter.disposed) {
398                    if (adapter.started) {
399                        adapter.stop(object);
400                    }
401                    adapter.components.remove(object);
402                    if (!adapter.disposed) {
403                        adapter.dispose(object);
404                    }
405                }
406                return result && !adapter.disposed;
407            }
408
409        }

411        /**
412         * Start of the container ensures that at least one pooled component has been started. Applies
413         * only if the delegated {@link com.picocontainer.ComponentAdapter} supports a lifecylce by implementing
414         * {@link com.picocontainer.LifecycleStrategy}.
415         *
416         * @throws IllegalStateException if pool was already disposed
417         */
418        @Override
419        public void start(final PicoContainer container) {
420            if (delegateHasLifecylce) {
421                if (started) {
422                                        throw new IllegalStateException("Already started");
423                                }
424                if (disposed) {
425                                        throw new IllegalStateException("Already disposed");
426                                }
427                for (Object component : components) {
428                    start(component);
429                }
430                started = true;
431                if (pool.size() == 0) {
432                    getComponentInstance(container, NOTHING.class);
433                }
434            }
435        }
436
437        /**
438         * Stop of the container has no effect for the pool. Applies only if the delegated
439         * {@link com.picocontainer.ComponentAdapter} supports a lifecylce by implementing {@link com.picocontainer.LifecycleStrategy}.
440         *
441         * @throws IllegalStateException if pool was already disposed
442         */
443        @Override
444        public void stop(final PicoContainer container) {
445            if (delegateHasLifecylce) {
446                if (!started) {
447                                        throw new IllegalStateException("Not started yet");
448                                }
449                if (disposed) {
450                                        throw new IllegalStateException("Already disposed");
451                                }
452                for (Object component : components) {
453                    stop(component);
454                }
455                started = false;
456            }
457        }
458
459        /**
460         * Dispose of the container will dispose all returning objects. They will not be added to the
461         * pool anymore. Applies only if the delegated {@link com.picocontainer.ComponentAdapter} supports a lifecylce by
462         * implementing {@link com.picocontainer.LifecycleStrategy}.
463         *
464         * @throws IllegalStateException if pool was already disposed
465         */
466        @Override
467        public void dispose(final PicoContainer container) {
468            if (delegateHasLifecylce) {
469                if (started) {
470                                        throw new IllegalStateException("Not stopped yet");
471                                }
472                if (disposed) {
473                                        throw new IllegalStateException("Already disposed");
474                                }
475                disposed = true;
476                for (Object component : components) {
477                    dispose(component);
478                }
479                // TODO: Release pooled components and clear collection
480            }
481        }

483        private synchronized void writeObject(final ObjectOutputStream out) throws IOException {
484            out.defaultWriteObject();
485            SerializationMode mode = serializationMode;
486            if (mode == SerializationMode.FORCE && components.size() > 0) {
487                try {
488                    final ByteArrayOutputStream buffer = new ByteArrayOutputStream();
489                    final ObjectOutputStream testStream = new ObjectOutputStream(buffer);
490                    testStream.writeObject(components); // force NotSerializableException
491                    testStream.close();
492                } catch (final NotSerializableException e) {
493                    mode = SerializationMode.NONE;
494                }
495            }
496            if (mode == SerializationMode.STANDARD) {
497                out.writeObject(components);
498            } else {
499                out.writeObject(new ArrayList());
500            }
501        }
502
503        private void readObject(final ObjectInputStream in) throws IOException, ClassNotFoundException {
504            in.defaultReadObject();
505            components = (List<Object>) in.readObject();
506        }
