154    public static class FooRunnable implements Runnable, Startable {
155        private int runCount;
156        private Thread thread = new Thread();
157        private boolean interrupted;
158
159        public FooRunnable() {
160        }
161
162        public int runCount() {
163            return runCount;
164        }
165
166        public boolean isInterrupted() {
167            return interrupted;
168        }
169
170        public void start() {
171            thread = new Thread(this);
172            thread.start();
173        }
174
175        public void stop() {
176            thread.interrupt();
177        }
178
179        // this would do something a bit more concrete
180        // than counting in real life !
181        public void run() {
182            runCount++;
183            try {
184                Thread.sleep(10000);
185            } catch (InterruptedException e) {
186                interrupted = true;
187            }
188        }
189    }

191    @Test public void testStartStopOfDaemonizedThread() throws Exception {
192        DefaultPicoContainer pico = new DefaultPicoContainer(new Caching());
193        pico.addComponent(FooRunnable.class);
194

196        pico.start();
197        Thread.sleep(100);
198        pico.stop();

202        pico.start();
203        Thread.sleep(100);

405    @Test public void testLifecycleCanRecoverWithCustomComponentMonitor() throws NoSuchMethodException {
406
407        final Startable s1 = mockery.mock(Startable.class, "s1");
408        final Startable s2 = mockery.mock(Startable.class, "s2");
409        final ComponentMonitor cm = mockery.mock(ComponentMonitor.class);
410        mockery.checking(new Expectations() {{
411            one(s1).start();
412            will(throwException(new RuntimeException("I do not want to start myself")));
413            one(s1).stop();
414            one(s2).start();
415            one(s2).stop();
416            // s1 expectations
417            one(cm).invoking(with(aNull(PicoContainer.class)), with(aNull(ComponentAdapter.class)), with(equal(Startable.class.getMethod("start", (Class[])null))), with(same(s1)), with(any(Object[].class)));
418            one(cm).lifecycleInvocationFailed(with(aNull(MutablePicoContainer.class)), with(aNull(ComponentAdapter.class)), with(any(Method.class)), with(same(s1)), with(any(RuntimeException.class)));
419            one(cm).invoking(with(aNull(PicoContainer.class)),
420                    with(aNull(ComponentAdapter.class)),
421                    with(equal(Startable.class.getMethod("stop", (Class[])null))),
422                    with(same(s1)), with(any(Object[].class)));
423            one(cm).invoked(with(aNull(PicoContainer.class)),
424                    with(aNull(ComponentAdapter.class)),
425                    with(equal(Startable.class.getMethod("stop", (Class[])null))),
426                    with(same(s1)), with(any(Long.class)), with(same(null)), with(any(Object[].class)));
427            // s2 expectations
428            one(cm).invoking(with(aNull(PicoContainer.class)), with(aNull(ComponentAdapter.class)), with(equal(Startable.class.getMethod("start", (Class[])null))), with(same(s2)), with(any(Object[].class)));
429            one(cm).invoked(with(aNull(PicoContainer.class)), with(aNull(ComponentAdapter.class)), with(equal(Startable.class.getMethod("start", (Class[])null))), with(same(s2)), with(any(Long.class)), with(same(null)), with(any(Object[].class)));
430            one(cm).invoking(with(aNull(PicoContainer.class)), with(aNull(ComponentAdapter.class)), with(equal(Startable.class.getMethod("stop", (Class[])null))), with(same(s2)), with(any(Object[].class)));
431            one(cm).invoked(with(aNull(PicoContainer.class)), with(aNull(ComponentAdapter.class)), with(equal(Startable.class.getMethod("stop", (Class[])null))), with(same(s2)), with(any(Long.class)), with(same(null)), with(any(Object[].class)));
432        }});
433
434        DefaultPicoContainer dpc = new DefaultPicoContainer(cm);
435        dpc.addComponent("foo", s1);
436        dpc.addComponent("bar", s2);
437        dpc.start();
438        dpc.stop();
439    }

441    @Test public void testLifecycleFailuresCanBePickedUpAfterTheEvent() {
442        final Startable s1 = mockery.mock(Startable.class, "s1");
443        final Startable s2 = mockery.mock(Startable.class, "s2");
444        final Startable s3 = mockery.mock(Startable.class, "s3");
445        mockery.checking(new Expectations() {{
446            one(s1).start();
447            will(throwException(new RuntimeException("I do not want to start myself")));
448            one(s1).stop();
449            one(s2).start();
450            one(s2).stop();
451            one(s3).start();
452            will(throwException(new RuntimeException("I also do not want to start myself")));
453            one(s3).stop();
454        }});
455
456        LifecycleComponentMonitor lifecycleComponentMonitor = new LifecycleComponentMonitor(new NullComponentMonitor());
457
458        DefaultPicoContainer dpc = new DefaultPicoContainer(lifecycleComponentMonitor);
459        dpc.addComponent("one", s1);
460        dpc.addComponent("two", s2);
461        dpc.addComponent("three", s3);
462
463        dpc.start();
464
465        try {
466            lifecycleComponentMonitor.rethrowLifecycleFailuresException();
467            fail("LifecycleFailuresException expected");
468        } catch (LifecycleFailuresException e) {
469            assertEquals("I do not want to start myself;  I also do not want to start myself;", e.getMessage().trim());
470            dpc.stop();
471            assertEquals(2, e.getFailures().size());
472        }
473
474    }
