295    @Test public void testMonitoringHappensBeforeAndAfterInstantiation() throws NoSuchMethodException {
296        final ComponentMonitor monitor = mockery.mock(ComponentMonitor.class);
297        final Constructor emptyHashMapCtor = HashMap.class.getConstructor();
298        final Matcher<Long> durationIsGreaterThanOrEqualToZero = new BaseMatcher<Long>() {
299                public boolean matches(final Object item) {
300                Long duration = (Long)item;
301                return 0 <= duration;
302                        }
303
304                        public void describeTo(final Description description) {
305                description.appendText("The endTime wasn't after the startTime");
306                        }
307        };
308
309        final Matcher<Object> isAHashMapThatWozCreated = new BaseMatcher<Object>() {
310                public boolean matches(final Object item) {
311                return item instanceof HashMap;
312            }
313
314                        public void describeTo(final Description description) {
315                description.appendText("Should have been a hashmap");
316                        }
317        };
318
319        final Matcher<Object[]> injectedIsEmptyArray = new BaseMatcher<Object[]>() {
320                public boolean matches(final Object item) {
321                Object[] injected = (Object[])item;
322                return 0 == injected.length;
323            }
324                public void describeTo(final Description description) {
325                description.appendText("Should have had nothing injected into it");
326            }
327        };
328
329        mockery.checking(new Expectations() {{
330                one(monitor).instantiating(with(any(PicoContainer.class)), (ComponentAdapter)with(a(ConstructorInjection.ConstructorInjector.class)), with(equal(emptyHashMapCtor)));
331                will(returnValue(emptyHashMapCtor));
332                one(monitor).instantiated(with(any(PicoContainer.class)), (ComponentAdapter)with(a(ConstructorInjection.ConstructorInjector.class)), with(equal(emptyHashMapCtor)),
333                                with(isAHashMapThatWozCreated), with(injectedIsEmptyArray),
334                                with(durationIsGreaterThanOrEqualToZero));
335        }});
336
337        ConstructorInjection.ConstructorInjector cica = new ConstructorInjection.ConstructorInjector(monitor, false,
338                Map.class, HashMap.class, new ConstructorParameters(new Parameter[0]));
339        cica.getComponentInstance(null, ComponentAdapter.NOTHING.class);
340    }

403    @Test public void testSpeedOfRememberedConstructor()  {
404        long with, without;
405
406        ConstructorInjection injectionType = new ForgetfulConstructorInjection();
407        timeIt(injectionType, 10); // discard
408        timeIt(injectionType, 10); // discard
409        timeIt(injectionType, 10); // discard
410        without = timeIt(injectionType, 20000);
411        injectionType = new ConstructorInjection();
412        garbageCollect();
413        timeIt(injectionType, 10); // discard
414        timeIt(injectionType, 10); // discard
415        timeIt(injectionType, 10); // discard
416        with = timeIt(injectionType, 20000);
417        assertTrue("'with' should be less than 'without' but they were in fact with: " + with + ", and without:" + without, with < without);
418    }
419
420        private void garbageCollect() throws Error {
421                Runtime rt = Runtime.getRuntime();
422        rt.gc();
423        rt.runFinalization();
424        final CountDownLatch latch = new CountDownLatch(1);
425        new Object() {
426          @Override
427                protected void finalize() {
428            latch.countDown();
429          }
430        };
431        rt.gc();
432        rt.runFinalization();
433        try {
434          latch.await();
435        }
436        catch(InterruptedException ie){
437          throw new Error(ie);
438        }
439        }

441    private long timeIt(final ConstructorInjection injectionType, final int iterations) {
442        DefaultPicoContainer dpc = new DefaultPicoContainer(injectionType);
443        Two two = new Two();
444        dpc.addComponent(two);
445        dpc.addComponent(One.class);
446        long start = System.currentTimeMillis();
447        for (int x = 0; x < iterations; x++) {
448                dpc.getComponent(One.class);
449            }
450        long end = System.currentTimeMillis();
451        assertEquals(iterations, two.howMany());
452        return end-start;
453    }
