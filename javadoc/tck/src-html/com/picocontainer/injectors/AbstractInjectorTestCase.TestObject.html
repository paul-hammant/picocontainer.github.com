050    @Test public void testCaughtIllegalAccessExceptionInvokesMonitorAndThrows() {
051        final EmptyPicoContainer epc = new EmptyPicoContainer();
052        final IllegalAccessException iae = new IllegalAccessException("foo");
053        NullComponentMonitor ncm = new NullComponentMonitor() {
054            @Override
055                        public void instantiationFailed(final PicoContainer container,
056                                            final ComponentAdapter componentAdapter,
057                                            final Constructor constructor,
058                                            final Exception e) {
059                assertSame(epc, container);
060                assertSame(ai, componentAdapter);
061                assertSame(ctor, constructor);
062                assertSame(iae, e);
063            }
064        };
065        try {
066            ai.caughtIllegalAccessException(ncm, ctor, iae, epc);
067        } catch (PicoCompositionException e) {
068            assertSame(iae, e.getCause());
069        }
070    }

097            @Override
098                        public void invocationFailed(final Member member, final Object instance, final Exception e) {

103        };
104        try {
105            ai.caughtInvocationTargetException(ncm, ctor, "bar", ite);

184    @Test
185    public void testAmbiguousComponentExceptionMessageWithNoAccsesibleObjectOrNoComponent() {
186        Generic<String> generic = Generic.get(String.class);
187        Object[] keys = new Object[] {String.class, "fwibble", "fribbit"};
188        AmbiguousComponentResolutionException acre = new AmbiguousComponentResolutionException(generic, keys);
189        assertTrue("Got " + acre.getMessage(),
190                        acre.getMessage().contains("<not-specified> needs a 'java.lang.String' injected through : <unknown>'," +
191                                        " but there are too many choices to inject. These:[class java.lang.String, fwibble, fribbit]"));
