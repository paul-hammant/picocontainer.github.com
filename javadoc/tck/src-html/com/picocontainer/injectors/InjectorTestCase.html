018/**
019 * test capabilities of injector factory. as this is mostly convenience wrapper around
020 * constructors, we just test that everything was passed through
021 *
022 * @author ko5tik
023 */
024public class InjectorTestCase {
025
026    final Object key = new Object();
027    final Parameter checked = new ConstantParameter(null);
028    final Parameter[] checkedArray = new Parameter[]{checked};
029    final ComponentMonitor monitor = new NullComponentMonitor();
030
031
032    final PicoVisitor parameterChecker = new AbstractPicoVisitor() {
033
034
035        public boolean visitContainer(final PicoContainer pico) {
036            return false;  //To change body of implemented methods use File | Settings | File Templates.
037        }
038
039        public void visitComponentAdapter(final ComponentAdapter<?> componentAdapter) {
040            //To change body of implemented methods use File | Settings | File Templates.
041        }
042
043        public void visitComponentFactory(final ComponentFactory componentFactory) {
044            //To change body of implemented methods use File | Settings | File Templates.
045        }
046
047        public void visitParameter(final Parameter parameter) {
048            assertSame(checked, parameter);
049        }
050    };
051
052    /**
053     * test that all parameters were passed to respective constructor
054     */
055    @Test
056    public void testSimpleConstructor() {
057        ComponentAdapter adapter = constructor(key, getClass(), checked);
058        assertSame(key, adapter.getComponentKey());
059        assertSame(getClass(), adapter.getComponentImplementation());
060        adapter.accept(parameterChecker);
061    }
062}
