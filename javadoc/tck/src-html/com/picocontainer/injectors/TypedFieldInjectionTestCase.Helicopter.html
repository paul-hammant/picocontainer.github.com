010package com.picocontainer.injectors;
011
012import static org.junit.Assert.assertEquals;
013import static org.junit.Assert.assertTrue;
014
015import java.util.HashMap;
016import java.util.Map;
017import java.util.Properties;
018
019import org.junit.Test;
020
021import com.picocontainer.ComponentAdapter;
022import com.picocontainer.injectors.TypedFieldInjection;
023import com.picocontainer.lifecycle.NullLifecycleStrategy;
024import com.picocontainer.monitors.ConsoleComponentMonitor;
025
026public class TypedFieldInjectionTestCase {
027    private static final String FIELD_TYPES = Integer.class.getName() + " " + PogoStick.class.getName() + " " + Float.class.getName();
028
029    public static class Helicopter {
030        private PogoStick pogo;
031    }
032
033    public static class PogoStick {
034    }
035
036
037    @Test public void testFactoryMakesNamedInjector() {
038
039        TypedFieldInjection injectionFactory = new TypedFieldInjection();
040
041        ConsoleComponentMonitor cm = new ConsoleComponentMonitor();
042        Properties props = new Properties();
043        props.setProperty("injectionFieldTypes", FIELD_TYPES);
044        ComponentAdapter ca = injectionFactory.createComponentAdapter(cm, new NullLifecycleStrategy(),
045                props, Map.class, HashMap.class, null, null, null);
046
047        assertTrue(ca instanceof TypedFieldInjection.TypedFieldInjector);
048
049        TypedFieldInjection.TypedFieldInjector tfi = (TypedFieldInjection.TypedFieldInjector) ca;
050
051        assertEquals(3, tfi.getInjectionFieldTypes().size());
052        assertEquals(Integer.class.getName(), tfi.getInjectionFieldTypes().get(0));
053        assertEquals(PogoStick.class.getName(), tfi.getInjectionFieldTypes().get(1));
054        assertEquals(Float.class.getName(), tfi.getInjectionFieldTypes().get(2));
055    }
056
057    @Test public void testPropertiesAreRight() {
058        Properties props = TypedFieldInjection.injectionFieldTypes(FIELD_TYPES);
059        assertEquals("java.lang.Integer com.picocontainer.injectors.TypedFieldInjectionTestCase$PogoStick java.lang.Float", props.getProperty("injectionFieldTypes"));
060        assertEquals(1, props.size());
061    }
062
063
064}
