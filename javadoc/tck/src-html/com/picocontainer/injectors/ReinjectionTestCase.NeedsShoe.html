072    public static class NeedsShoe implements INeedsShoe {
073        private final Shoe bar;
074        private String string;
075
076        public NeedsShoe(final Shoe bar) {
077            this.bar = bar;
078        }
079
080        @Hurrah
081        public int doIt(final String s) {
082            this.string = s;
083            return Integer.parseInt(s) / 2;
084        }
085
086        public int doInt(final int s) {
087            this.string = "i="+ s;
088            return s/2;
089        }
090
091        public Object getBar() {
092            return bar;
093        }
094
095        public Object getString() {
096            return string;
097        }
098        public static enum M {
099            doIt("doIt", String.class);
100            private Method method;
101
102            M(final String s, final Class... paramTypes) {
103                try {
104                    method = NeedsShoe.class.getMethod(s, paramTypes);
105                } catch (NoSuchMethodException e) {
106                    throw new UnsupportedOperationException(e);
107                }
108            }
109            public Method toMethod() {
110                return method;
111            }
112        }
113    }

140    private void cachedComponentCanBeReinjectedByATransientChildContainer(final AbstractInjectionType methodInjection) {
141        DefaultPicoContainer parent = new DefaultPicoContainer(new Caching().wrap(new ConstructorInjection()));
142        parent.addComponent(INeedsShoe.class, NeedsShoe.class);
143        parent.addComponent(Shoe.class);
144        parent.addComponent("12");
145
146        INeedsShoe needsShoe = parent.getComponent(INeedsShoe.class);
147        assertNotNull(needsShoe.getBar());
148        assertTrue(needsShoe.getString() == null);
149
150        TransientPicoContainer tpc = new TransientPicoContainer(new Reinjection(methodInjection, parent), parent);
151        tpc.addComponent(INeedsShoe.class, NeedsShoe.class);
152
153        INeedsShoe needsShoe2 = tpc.getComponent(INeedsShoe.class);
154        assertSame(needsShoe, needsShoe2);
155        assertNotNull(needsShoe2.getBar());
156        assertNotNull(needsShoe2.getString());
157
158        INeedsShoe needsShoe3 = parent.getComponent(INeedsShoe.class);
159        assertSame(needsShoe, needsShoe3);
160        assertNotNull(needsShoe3.getBar());
161        assertNotNull(needsShoe3.getString());
162    }

164    @Test
165    public void confirmThatReinjectionCanLeverageParameterNamesForDisambiguation() {
166        MethodInjection methodInjection = new MethodInjection(DOIT_METHOD);
167        DefaultPicoContainer parent = new DefaultPicoContainer(new Caching().wrap(new ConstructorInjection()));
168
169        // parameter name leverage can't work on interfaces if using bytecode retrieval technique
170
171        parent.addComponent(NeedsShoe.class);
172        parent.addComponent(Shoe.class);
173        parent.addComponent("a", "1333");
174        parent.addComponent("s", "12");
175        parent.addComponent("tjklhjkjhkjh", "44");
176
177        NeedsShoe needsShoe = parent.getComponent(NeedsShoe.class);
178        assertNotNull(needsShoe.bar);
179        assertTrue(needsShoe.string == null);
180
181        Reinjection reinjection = new Reinjection(methodInjection, parent);
182        TransientPicoContainer tpc = new TransientPicoContainer(reinjection, parent);
183        tpc.as(Characteristics.USE_NAMES).addComponent(NeedsShoe.class);
184
185        NeedsShoe needsShoe2 = tpc.getComponent(NeedsShoe.class);
186        assertSame(needsShoe, needsShoe2);
187        assertNotNull(needsShoe2.bar);
188        assertNotNull(needsShoe2.string);
189        assertEquals("12", needsShoe2.string);
190
191    }

234    public static class ReturnParameterAction implements Action {
235        private final int parameter;
236
237        public ReturnParameterAction(final int parameter) {
238            this.parameter = parameter;
239        }
240
241        public void describeTo(final Description description) {
242            // describe it
243        }
244
245        public Object invoke(final Invocation invocation) {
246            return invocation.getParameter(parameter);
247        }
248    }

250    private void cachedComponentCanBeReinjectedByATransientReinjector(final AbstractInjectionType methodInjection) {
251        final DefaultPicoContainer parent = new DefaultPicoContainer(new Caching().wrap(new ConstructorInjection()));
252        parent.setName("parent");
253        parent.addComponent(INeedsShoe.class, NeedsShoe.class);
254        parent.addComponent(Shoe.class);
255        parent.addComponent("12");
256
257        final INeedsShoe foo = parent.getComponent(INeedsShoe.class);
258        assertNotNull(foo.getBar());
259        assertTrue(foo.getString() == null);
260
261        final ComponentMonitor cm = mockery.mock(ComponentMonitor.class);
262        Reinjector reinjector = new Reinjector(parent, cm);
263        mockery.checking(new Expectations() {{
264            atLeast(1).of(cm).newInjector(with(any(com.picocontainer.Injector.class)));
265            will(new ReturnParameterAction(0));
266            one(cm).invoking(with(any(PicoContainer.class)), with(any(ComponentAdapter.class)),
267                    with(any(Method.class)), with(any(Object.class)), with(any(Object[].class)));
268            will(returnValue(ComponentMonitor.KEEP));
269            one(cm).invoked(with(any(PicoContainer.class)), with(any(ComponentAdapter.class)),
270                    with(any(Method.class)), with(any(Object.class)), with(any(Long.class)), with(any(Integer.class)), with(any(Object[].class)));
271        }});
272
273        Object o = reinjector.reinject(NeedsShoe.class, methodInjection);
274        int result = (Integer) o;
275        assertEquals(6, result);
276
277        INeedsShoe foo3 = parent.getComponent(INeedsShoe.class);
278        assertSame(foo, foo3);
279        assertNotNull(foo3.getBar());
280        assertNotNull(foo3.getString());
281        assertEquals("12", foo3.getString());
282    }
