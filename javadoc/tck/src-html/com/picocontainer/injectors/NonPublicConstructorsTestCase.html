009package com.picocontainer.injectors;
010
011import static junit.framework.Assert.assertNotNull;
012
013import org.junit.Test;
014
015import com.picocontainer.ComponentAdapter;
016import com.picocontainer.DefaultPicoContainer;
017import com.picocontainer.MutablePicoContainer;
018import com.picocontainer.injectors.ConstructorInjection.ConstructorInjector;
019
020public final class NonPublicConstructorsTestCase {
021
022    @Test
023   public void doFirstSampleWithNotPublicConstructor() {
024        MutablePicoContainer container = new DefaultPicoContainer();
025
026        ComponentAdapter<DummyNotPublicConstructor> dummyComponentAdapter =
027                new ConstructorInjector<DummyNotPublicConstructor>(
028                        DummyNotPublicConstructor.class,
029                        DummyNotPublicConstructor.class)
030                .withNonPublicConstructors();
031
032        container.addAdapter(dummyComponentAdapter);
033
034        DummyNotPublicConstructor dummy = container.getComponent(DummyNotPublicConstructor.class);
035        assertNotNull(dummy);
036    }
037
038    @Test
039    public void doSecondSampleWithNotPublicClass() {
040        MutablePicoContainer container = new DefaultPicoContainer();
041
042        ComponentAdapter<DummyNotPublicClass> dummyComponentAdapter =
043                new ConstructorInjector<DummyNotPublicClass>(
044                        DummyNotPublicClass.class.getCanonicalName(),
045                        DummyNotPublicClass.class)
046                .withNonPublicConstructors();
047
048        container.addAdapter(dummyComponentAdapter);
049
050        Object item = container.getComponent(DummyNotPublicClass.class);
051        assertNotNull(item);
052    }
053
054    @Test
055    public void doThirdSampleWithProtectedConstructor() {
056        MutablePicoContainer container = new DefaultPicoContainer();
057
058        ComponentAdapter<DummyProtectedConstructor> dummyComponentAdapter =
059                new ConstructorInjector<DummyProtectedConstructor>(
060                        DummyProtectedConstructor.class,
061                        DummyProtectedConstructor.class)
062                .withNonPublicConstructors();
063
064
065        container.addAdapter(dummyComponentAdapter);
066
067        DummyProtectedConstructor dummy = container.getComponent(DummyProtectedConstructor.class);
068        assertNotNull(dummy);
069
070    }
071
072    public static class DummyProtectedConstructor {
073        protected DummyProtectedConstructor() {
074        }
075    }
076
077    public static class DummyNotPublicConstructor {
078        DummyNotPublicConstructor() {
079        }
080    }
081
082    static class DummyNotPublicClass {
083    }
084
085}
