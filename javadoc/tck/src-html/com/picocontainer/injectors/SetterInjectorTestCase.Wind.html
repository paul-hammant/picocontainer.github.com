361        SetterInjection.SetterInjector<A> aAdapter = new SetterInjection.SetterInjector<A>("a", A.class, new NullComponentMonitor(), "set", false, "", false, null
362       );
363        SetterInjection.SetterInjector<B> bAdapter = new SetterInjection.SetterInjector<B>("b", B.class, new NullComponentMonitor(), "set", false, "", false, null
364       );
365
366        MutablePicoContainer pico = new DefaultPicoContainer();
367        pico.setName("parent");
368        pico.addAdapter(bAdapter)
369                .addAdapter(aAdapter);
370
371        try {
372            aAdapter.getComponentInstance(pico, ComponentAdapter.NOTHING.class);
373        } catch (AbstractInjector.UnsatisfiableDependenciesException e) {
374            assertNotNull(e.getMessage());
375            String message = e.getMessage().replace("com.picocontainer.injectors.SetterInjectorTestCase$", "");
376            message.trim();
377
378            System.out.println("-----------  Message -----------------");
379            System.out.println(message);
380            System.out.println("-----------==========-----------------");
381
382            //Order can't be determined so we "standardize" the order of things so to speak.
383            message = message.replaceAll(" ","");
384            message = message.replaceAll("\t","");
385            message = message.replace("interfacejava.util.List,classjava.lang.String",
386                                           "classjava.lang.String,interfacejava.util.List");
387            assertFalse(message.contains(" "));
388            message = message.replace(
389                                          "publicvoidA.setList(java.util.List),publicvoidA.setString(java.lang.String)",
390                                          "publicvoidA.setString(java.lang.String),publicvoidA.setList(java.util.List)"
391
392                                                        );
393
394
395            assertEquals("Got " + message, "Ahasunsatisfieddependencies[classjava.lang.String,interfacejava.util.List]formembers[publicvoidA.setString(java.lang.String),publicvoidA.setList(java.util.List)]fromparent:2<|",
396                    message);
397        }

478        final Mockery mockery = new Mockery();
479        final ComponentMonitor componentMonitor = mockery.mock(ComponentMonitor.class);
480        final MutablePicoContainer pico = new DefaultPicoContainer(new EmptyPicoContainer(),
481             new NullLifecycleStrategy(), componentMonitor, new SetterInjection());

484            oneOf(componentMonitor).newInjector(with(any(com.picocontainer.Injector.class)));

486        }});

488        pico.addComponent(B.class);
489
