056        public static class ReadAllowedAccessControllerWrapper implements AccessControllerWrapper {
057                public void checkPermission(Permission checkingPermission) throws AccessControlException {
058                        if (readPermission.implies(checkingPermission)) {
059                                return;
060                        }
061                        
062                        throw new AccessControlException("Write not allowed");
063                }
064        }

066        public static class WriteAllowedAccessControllerWrapper implements AccessControllerWrapper {
067                public void checkPermission(Permission checkingPermission) throws AccessControlException {
068                        if (writePermission.implies(checkingPermission)) {
069                                return;
070                        }
071                        
072                        throw new AccessControlException("Write not allowed");
073                }
074        }

076        public static class ReadWriteAllowedAccessControllerWrapper implements AccessControllerWrapper {
077                public void checkPermission(Permission checkingPermission) throws AccessControlException {
078                        if (readPermission.implies(checkingPermission)  || writePermission.implies(checkingPermission)) {

517        private void expectSuccess(String methodName, Object[] args, Class<?>... argTypes) throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
518                invokeMethod(methodName, args, argTypes);
519        }

521        
522        private void expectFailure(String methodName, Object[] args, Class<?>... argTypes) throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
523                try {
524                        invokeMethod(methodName, args, argTypes);
525                        fail("Expected Failure");
526                } catch (InvocationTargetException e) {
527                        if (e.getCause() instanceof AccessControlException) {
528                                assertNotNull(e.getCause().getMessage());
529                        } else {
530                                throw e;
531                        }
532                }
533        }
534        
535        private void invokeMethod(String methodName, Object[] args, Class<?>... argTypes) throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
536                Method m = pico.getClass().getMethod(methodName, argTypes);
537                m.invoke(pico, args);
538        }
