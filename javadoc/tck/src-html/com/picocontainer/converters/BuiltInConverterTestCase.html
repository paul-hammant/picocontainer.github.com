001package com.picocontainer.converters;
002
003import static junit.framework.Assert.assertFalse;
004import static junit.framework.Assert.assertTrue;
005import static org.junit.Assert.assertEquals;
006
007import java.io.File;
008import java.net.MalformedURLException;
009import java.net.URL;
010
011import javax.swing.JPanel;
012
013import org.junit.Test;
014
015import com.picocontainer.converters.BuiltInConverters;
016import com.picocontainer.converters.Converter;
017
018public class BuiltInConverterTestCase {
019    BuiltInConverters bic = new BuiltInConverters();
020
021    @Test
022    public void canConvertAllPrimitiveTypes() {
023        assertTrue(bic.canConvert(Integer.class));
024        assertTrue(bic.canConvert(Integer.TYPE));
025        assertTrue(bic.canConvert(Long.class));
026        assertTrue(bic.canConvert(Long.TYPE));
027        assertTrue(bic.canConvert(Byte.class));
028        assertTrue(bic.canConvert(Byte.TYPE));
029        assertTrue(bic.canConvert(Double.class));
030        assertTrue(bic.canConvert(Double.TYPE));
031        assertTrue(bic.canConvert(Float.class));
032        assertTrue(bic.canConvert(Float.TYPE));
033        assertTrue(bic.canConvert(Character.class));
034        assertTrue(bic.canConvert(Character.TYPE));
035        assertTrue(bic.canConvert(Short.class));
036        assertTrue(bic.canConvert(Short.TYPE));
037        assertTrue(bic.canConvert(Boolean.class));
038        assertTrue(bic.canConvert(Boolean.TYPE));
039
040        assertEquals(12, bic.convert("12", Integer.class));
041        assertEquals(12, bic.convert("12", Integer.TYPE));
042        assertEquals(12345678901L, bic.convert("12345678901",Long.TYPE));
043        assertEquals(12345678901L, bic.convert("12345678901",Long.class));
044        assertEquals((byte)12, bic.convert("12", Byte.class));
045        assertEquals((byte)12, bic.convert("12", Byte.TYPE));
046        assertEquals(2.22, bic.convert("2.22", Double.class));
047        assertEquals(2.22, bic.convert("2.22", Double.TYPE));
048        assertEquals(1.11F, bic.convert("1.11", Float.class));
049        assertEquals(1.11F, bic.convert("1.11", Float.TYPE));
050        assertEquals('a', bic.convert("a", Character.class));
051        assertEquals('a', bic.convert("a", Character.TYPE));
052        assertEquals((short)12, bic.convert("12", Short.class));
053        assertEquals((short)12, bic.convert("12", Short.TYPE));
054        assertEquals(Boolean.TRUE, bic.convert("TRUE", Boolean.class));
055        assertEquals(true, bic.convert("TRUE", Boolean.TYPE));
056    }
057
058
059    @Test
060    public void canConvertFileAndURL() throws MalformedURLException {
061        assertTrue(bic.canConvert(File.class));
062        assertTrue(bic.canConvert(URL.class));
063        assertEquals(new File("c:\\foo"), bic.convert("c:\\foo", File.class));
064        assertEquals(new URL("http://example.com"), bic.convert("http://example.com", URL.class));
065    }
066
067    @Test
068    public void canAddCustomConverter() {
069        bic.addConverter(new JPanelConverter(), JPanel.class);
070        assertTrue(bic.convert("anything", JPanel.class) instanceof JPanel);
071    }
072
073    @SuppressWarnings("serial")
074    @Test
075    public void canSupplementBuiltInConverters() {
076        assertFalse(bic.canConvert(JPanel.class));
077
078        BuiltInConverters bicWithJPanel = new BuiltInConverters() {
079            @Override
080            protected void addBuiltInConverters() {
081                super.addBuiltInConverters();
082                super.addConverter(new JPanelConverter(), JPanel.class);
083            }
084        };
085        assertEquals(Boolean.TRUE, bicWithJPanel.convert("TRUE", Boolean.class));
086        assertTrue(bicWithJPanel.convert("anything", JPanel.class) instanceof JPanel);
087
088    }
089
090    private static class JPanelConverter implements Converter<JPanel> {
091        public JPanel convert(final String paramValue) {
092            return new JPanel();
093        }
094    }
095
096
097}
