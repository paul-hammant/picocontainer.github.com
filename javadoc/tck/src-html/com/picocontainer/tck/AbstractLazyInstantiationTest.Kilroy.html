010package com.picocontainer.tck;
011
012import static org.junit.Assert.assertEquals;
013import static org.junit.Assert.assertNotNull;
014import static org.junit.Assert.assertSame;
015
016import org.junit.Test;
017
018import com.picocontainer.Characteristics;
019import com.picocontainer.MutablePicoContainer;
020import com.picocontainer.PicoException;
021
022/**
023 * @author Aslak Helles&oslash;y
024 */
025public abstract class AbstractLazyInstantiationTest {
026
027    protected abstract MutablePicoContainer createPicoContainer();
028
029    public static class Kilroy {
030        public Kilroy(final Havana havana) {
031            havana.graffiti("Kilroy was here");
032        }
033    }
034
035    public static class Havana {
036        public String paint = "Clean wall";
037
038        public void graffiti(final String paint) {
039            this.paint = paint;
040        }
041    }
042
043    @Test public void testLazyInstantiation() throws PicoException {
044        MutablePicoContainer pico = createPicoContainer();
045
046        pico.as(Characteristics.CACHE).addComponent(Kilroy.class);
047        pico.as(Characteristics.CACHE).addComponent(Havana.class);
048
049        assertSame(pico.getComponent(Havana.class), pico.getComponent(Havana.class));
050        assertNotNull(pico.getComponent(Havana.class));
051        assertEquals("Clean wall", pico.getComponent(Havana.class).paint);
052        assertNotNull(pico.getComponent(Kilroy.class));
053        assertEquals("Kilroy was here", pico.getComponent(Havana.class).paint);
054    }
055}
