295    @Test public void testComponentAdapterRegistrationOrderIsMaintained() throws NoSuchMethodException {
296
297        ConstructorInjection.ConstructorInjector c1 = new ConstructorInjection.ConstructorInjector<Object>("1", Object.class);
298        ConstructorInjection.ConstructorInjector c2 = new ConstructorInjection.ConstructorInjector<String>("2", String.class);
299
300        MutablePicoContainer picoContainer = createPicoContainer(null);
301        picoContainer.addAdapter(c1).addAdapter(c2);
302        Collection<ComponentAdapter<?>> list2 = picoContainer.getComponentAdapters();
303        //registration order should be maintained
304        assertEquals(2, list2.size());
305        assertEquals(c1.getComponentKey(), ((ComponentAdapter)list2.toArray()[0]).getComponentKey());
306        assertEquals(c2.getComponentKey(), ((ComponentAdapter)list2.toArray()[1]).getComponentKey());
307
308        picoContainer.getComponents(); // create all the instances at once
309        assertFalse("instances should be created in same order as adapters are created",
310                    picoContainer.getComponents().get(0) instanceof String);
311        assertTrue("instances should be created in same order as adapters are created",
312                   picoContainer.getComponents().get(1) instanceof String);
313
314        MutablePicoContainer reversedPicoContainer = createPicoContainer(null);
315        reversedPicoContainer.addAdapter(c2);
316        reversedPicoContainer.addAdapter(c1);
317        //registration order should be maintained
318        list2 = reversedPicoContainer.getComponentAdapters();
319        assertEquals(2, list2.size());
320        assertEquals(c2.getComponentKey(), ((ComponentAdapter)list2.toArray()[0]).getComponentKey());
321        assertEquals(c1.getComponentKey(), ((ComponentAdapter)list2.toArray()[1]).getComponentKey());
322
323        reversedPicoContainer.getComponents(); // create all the instances at once
324        assertTrue("instances should be created in same order as adapters are created",
325                   reversedPicoContainer.getComponents().get(0) instanceof String);
326        assertFalse("instances should be created in same order as adapters are created",
327                    reversedPicoContainer.getComponents().get(1) instanceof String);
328    }
