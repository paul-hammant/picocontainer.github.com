001package com.picocontainer;
002
003import static org.junit.Assert.assertEquals;
004
005import javax.inject.Inject;
006
007import org.junit.Test;
008
009import com.picocontainer.DefaultPicoContainer;
010import com.picocontainer.MutablePicoContainer;
011
012public class Issue0388TestCase {
013
014        @Test
015         public void testComponentCanUseConstructorAndAtInjectWithoutAnInitMethod() {
016
017
018                final MutablePicoContainer pico = new DefaultPicoContainer();
019                pico.addComponent(Carrot.class)
020                        .addComponent(Cheese.class)
021                        .addComponent(Mixer.class);
022
023                // fails with com.picocontainer.PicoCompositionException: java.lang.NoSuchMethodException: ...$Mixer.<init>()
024                final Utensil u = pico.getComponent(Utensil.class);
025                assertEquals("Mixing carrot with some cheese", u.use());
026            }
027
028            public static final class Cheese {
029            }
030
031            public static final class Carrot {
032            }
033
034            public static interface Utensil {
035                String use();
036            }
037
038            public static class Spoon implements Utensil {
039                public String use() {
040                    return "there is no spoon";
041                }
042            }
043
044            public static class Mixer implements Utensil {
045                private final Cheese cheese;
046                @Inject
047                private Carrot veg;
048
049                public Mixer(final Cheese cheese) {
050                    this.cheese = cheese;
051                }
052
053                public String use() {
054                    return "Mixing " + simpleClassNameOrNull(veg) + " with some " + simpleClassNameOrNull(cheese);
055                }
056            }
057
058            private static String simpleClassNameOrNull(final Object o) {
059                return o == null ? null : o.getClass().getSimpleName().toLowerCase();
060            }
061
062}
