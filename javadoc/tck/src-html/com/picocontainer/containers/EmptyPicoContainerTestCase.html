011package com.picocontainer.containers;
012
013import static org.junit.Assert.assertNull;
014import static org.junit.Assert.assertTrue;
015
016import org.junit.Test;
017
018import com.googlecode.jtype.Generic;
019import com.picocontainer.NameBinding;
020import com.picocontainer.PicoContainer;
021import com.picocontainer.containers.EmptyPicoContainer;
022
023/**
024 * @author Mauro Talevi
025 */
026public class EmptyPicoContainerTestCase {
027
028    @Test public void testReturnValues() {
029        PicoContainer pico = new EmptyPicoContainer();
030        assertNull(pico.getComponentAdapter((Generic)null, (NameBinding) null));
031        assertNull(pico.getComponentAdapter((Generic)null, (NameBinding) null));
032        assertTrue(pico.getComponentAdapters().isEmpty());
033        assertTrue(pico.getComponentAdapters((Generic) null).isEmpty());
034        assertNull(pico.getComponent((Class) null));
035        assertNull(pico.getComponent((Class) null));
036        assertTrue(pico.getComponents().isEmpty());
037        assertTrue(pico.getComponents(null).isEmpty());
038        assertNull(pico.getParent());
039    }
040}
