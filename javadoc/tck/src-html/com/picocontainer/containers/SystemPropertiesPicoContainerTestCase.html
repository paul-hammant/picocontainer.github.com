001package com.picocontainer.containers;
002
003import static org.junit.Assert.assertEquals;
004import static org.junit.Assert.assertSame;
005
006import org.junit.Test;
007
008import com.picocontainer.DefaultPicoContainer;
009import com.picocontainer.containers.SystemPropertiesPicoContainer;
010
011/**
012 * test capabilities of system properties providing container.
013 * @author Konstantin Pribluda
014 *
015 */
016public class SystemPropertiesPicoContainerTestCase {
017
018
019        /**
020         * all the content of system properties shall be made available
021         *  through this contaienr.
022         */
023        @Test public void testThatAllSystemPropertiesAreCopied() {
024                SystemPropertiesPicoContainer container = new SystemPropertiesPicoContainer();
025                for(Object key: System.getProperties().keySet()) {
026                        assertSame(System.getProperties().get(key),container.getComponent(key));
027                }
028        }
029
030    @Test public void testRepresentationOfContainerTree() {
031        SystemPropertiesPicoContainer parent = new SystemPropertiesPicoContainer();
032        parent.setName("parent");
033        DefaultPicoContainer child = new DefaultPicoContainer(parent);
034        child.setName("child");
035                child.addComponent("hello", "goodbye");
036        child.addComponent("bonjour", "aurevior");
037        int num = System.getProperties().size();
038        assertEquals("child:2<[Immutable]:[SysProps]:parent:"+num+"<|", child.toString());
039    }
