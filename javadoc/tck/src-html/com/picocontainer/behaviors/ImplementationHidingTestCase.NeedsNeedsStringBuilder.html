036    @Test public void testAddComponentUsesImplementationHidingBehavior() {
037        DefaultPicoContainer pico =
038            new DefaultPicoContainer(new EmptyPicoContainer(), new NullLifecycleStrategy(), new ImplementationHiding().wrap(new ConstructorInjection()));
039        pico.addComponent("foo", String.class);
040        ComponentAdapter<?> foo = pico.getComponentAdapter("foo");
041        assertEquals(ImplementationHiding.HiddenImplementation.class, foo.getClass());
042        assertEquals(ConstructorInjection.ConstructorInjector.class, ((AbstractBehavior.AbstractChangedBehavior) foo).getDelegate().getClass());
043    }

098    public static interface NeedsStringBuilder {
099        void foo();
100    }
101    public static class NeedsStringBuilderImpl implements NeedsStringBuilder {
102        StringBuilder sb;
103
104        public NeedsStringBuilderImpl(final StringBuilder sb) {
105            this.sb = sb;
106            sb.append("<init>");
107        }
108        public void foo() {
109            sb.append("foo()");
110        }
111    }
112    public static class NeedsNeedsStringBuilder {
113
114        NeedsStringBuilder nsb;
115
116        public NeedsNeedsStringBuilder(final NeedsStringBuilder nsb) {
117            this.nsb = nsb;
118        }
119        public void foo() {
120            nsb.foo();
121        }
122    }

124    @Test public void testLazyInstantiationSideEffectWhenForceOfDelayedInstantiationOfDependantClass() {
125        DefaultPicoContainer pico =
126            new DefaultPicoContainer(new ImplementationHiding().wrap(new Caching().wrap(new ConstructorInjection())));
127        pico.addComponent(StringBuilder.class);
128        pico.addComponent(NeedsStringBuilder.class, NeedsStringBuilderImpl.class);
129        pico.addComponent(NeedsNeedsStringBuilder.class);
130        NeedsNeedsStringBuilder nnsb = pico.getComponent(NeedsNeedsStringBuilder.class);
131        assertNotNull(nnsb);
132        StringBuilder sb = pico.getComponent(StringBuilder.class);
133        assertEquals("", sb.toString()); // not instantiated yet
134        nnsb.foo();
135        assertEquals("<init>foo()", sb.toString()); // instantiated
136    }

138    @Test public void shouldInstantiateForEveryGetComponentCall() {
139        DefaultPicoContainer parent = new DefaultPicoContainer(new Caching());
140        parent.addComponent(StringBuilder.class);
141        DefaultPicoContainer pico =
142            new DefaultPicoContainer(parent, new ImplementationHiding());
143        pico.addComponent(NeedsStringBuilder.class, NeedsStringBuilderImpl.class);
144        NeedsStringBuilder nsb = pico.getComponent(NeedsStringBuilder.class);
145        nsb.foo();
146        nsb = pico.getComponent(NeedsStringBuilder.class);
147        nsb.foo();
148        StringBuilder sb = pico.getComponent(StringBuilder.class);
149        assertEquals("<init>foo()<init>foo()", sb.toString());
150    }
151
152    @Test public void shouldInstantiateForEveryMethodCall() {
153        cachingTestBody("<init>foo()foo()", new ImplementationHiding());
154    }
155
156    private void cachingTestBody(final String expectation, final ComponentFactory compFactory) {
157        DefaultPicoContainer parent = new DefaultPicoContainer(new Caching());
158        parent.addComponent(StringBuilder.class);
159        DefaultPicoContainer pico =
160            new DefaultPicoContainer(parent, compFactory);
161        pico.addComponent(NeedsStringBuilder.class, NeedsStringBuilderImpl.class);
162        NeedsStringBuilder nsb = pico.getComponent(NeedsStringBuilder.class);
163        assertNotNull(nsb);
164        nsb.foo();
165        nsb.foo();
166        StringBuilder sb = pico.getComponent(StringBuilder.class);
167        assertEquals(expectation, sb.toString());
168    }
