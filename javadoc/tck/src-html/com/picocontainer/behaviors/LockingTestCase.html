001package com.picocontainer.behaviors;
002
003import static com.picocontainer.Characteristics.LOCK;
004import static com.picocontainer.Characteristics.NO_LOCK;
005import static org.junit.Assert.assertNotNull;
006import static org.junit.Assert.assertNull;
007
008import org.junit.Test;
009import com.picocontainer.tck.AbstractComponentFactoryTest;
010
011import com.picocontainer.ComponentFactory;
012import com.picocontainer.MutablePicoContainer;
013import com.picocontainer.PicoBuilder;
014import com.picocontainer.behaviors.Locking;
015import com.picocontainer.injectors.AdaptingInjection;
016
017
018public class LockingTestCase extends AbstractComponentFactoryTest {
019
020        private final ComponentFactory locking = new Locking().wrap(new AdaptingInjection());
021
022
023        @Test
024        public void testPicocontainerPropertiesIntegration() {
025                MutablePicoContainer mpc = new PicoBuilder().withBehaviors(new Locking()).build();
026                mpc.as(LOCK).addComponent("locked","It is locked");
027                mpc.as(NO_LOCK).addComponent("not locked", "It is not locked");
028
029                assertNotNull(mpc.getComponentAdapter("locked").findAdapterOfType(Locking.Locked.class));
030                assertNull(mpc.getComponentAdapter("not locked").findAdapterOfType(Locking.Locked.class));
031
032        }
033
034        @Override
035        protected ComponentFactory createComponentFactory() {
036                return locking;
037        }
038
039}
