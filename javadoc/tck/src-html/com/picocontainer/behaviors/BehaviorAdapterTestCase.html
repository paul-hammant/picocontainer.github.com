010package com.picocontainer.behaviors;
011
012import static org.junit.Assert.assertEquals;
013import static org.junit.Assert.assertNotNull;
014import static org.junit.Assert.fail;
015import static com.picocontainer.tck.MockFactory.mockeryWithCountingNamingScheme;
016
017import org.jmock.Expectations;
018import org.jmock.Mockery;
019import org.jmock.integration.junit4.JMock;
020import org.junit.Test;
021import org.junit.runner.RunWith;
022import com.picocontainer.testmodel.SimpleTouchable;
023import com.picocontainer.testmodel.Touchable;
024
025import com.picocontainer.ChangedBehavior;
026import com.picocontainer.ComponentAdapter;
027import com.picocontainer.ComponentMonitor;
028import com.picocontainer.ComponentMonitorStrategy;
029import com.picocontainer.DefaultPicoContainer;
030import com.picocontainer.LifecycleStrategy;
031import com.picocontainer.PicoCompositionException;
032import com.picocontainer.PicoContainer;
033import com.picocontainer.behaviors.AbstractBehavior;
034
035/**
036 * @author Mauro Talevi
037 */
038@RunWith(JMock.class)
039@SuppressWarnings("serial")
040public class BehaviorAdapterTestCase {
041
042        private final Mockery mockery = mockeryWithCountingNamingScheme();
043
044    @Test public void testDecoratingComponentAdapterDelegatesToMonitorThatDoesSupportStrategy() {
045        AbstractBehavior.AbstractChangedBehavior adapter = new FooAbstractChangedBehavior(mockComponentAdapterThatDoesSupportStrategy());
046        adapter.changeMonitor(mockMonitorWithNoExpectedMethods());
047        assertNotNull(adapter.currentMonitor());
048    }
049
050    @Test public void testDecoratingComponentAdapterDelegatesToMonitorThatDoesNotSupportStrategy() {
051        AbstractBehavior.AbstractChangedBehavior adapter = new FooAbstractChangedBehavior(mockComponentAdapter());
052        adapter.changeMonitor(mockMonitorWithNoExpectedMethods());
053        try {
054            adapter.currentMonitor();
055            fail("PicoCompositionException expected");
056        } catch (PicoCompositionException e) {
057            assertEquals("No component monitor found in delegate", e.getMessage());
058        }
059    }
060
061    @Test public void testDecoratingComponentAdapterDelegatesLifecycleManagement() {
062        AbstractBehavior.AbstractChangedBehavior adapter = new FooAbstractChangedBehavior(mockComponentAdapterThatCanManageLifecycle());
063        PicoContainer pico = new DefaultPicoContainer();
064        adapter.start(pico);
065        adapter.stop(pico);
066        adapter.dispose(pico);
067        Touchable touchable = new SimpleTouchable();
068        adapter.start(touchable);
069        adapter.stop(touchable);
070        adapter.dispose(touchable);
071    }
072
073    @Test public void testDecoratingComponentAdapterIgnoresLifecycleManagementIfDelegateDoesNotSupportIt() {
074        AbstractBehavior.AbstractChangedBehavior adapter = new FooAbstractChangedBehavior(mockComponentAdapter());
075        PicoContainer pico = new DefaultPicoContainer();
076        adapter.start(pico);
077        adapter.stop(pico);
078        adapter.dispose(pico);
079        Touchable touchable = new SimpleTouchable();
080        adapter.start(touchable);
081        adapter.stop(touchable);
082        adapter.dispose(touchable);
083    }
084
085    ComponentMonitor mockMonitorWithNoExpectedMethods() {
086        return mockery.mock(ComponentMonitor.class);
087    }
088
089    private ComponentAdapter mockComponentAdapterThatDoesSupportStrategy() {
090        final ComponentAdapterThatSupportsStrategy ca = mockery.mock(ComponentAdapterThatSupportsStrategy.class);
091        mockery.checking(new Expectations() {{
092                one(ca).changeMonitor(with(any(ComponentMonitor.class)));
093                one(ca).currentMonitor();
094                will(returnValue(mockMonitorWithNoExpectedMethods()));
095        }});
096        return ca;
097    }
098
099    private ComponentAdapter mockComponentAdapter() {
100         return mockery.mock(ComponentAdapter.class);
101    }
102
103    public static interface ComponentAdapterThatSupportsStrategy extends ComponentAdapter, ComponentMonitorStrategy {
104    }
105
106    private ComponentAdapter mockComponentAdapterThatCanManageLifecycle() {
107        final ComponentAdapterThatCanManageLifecycle ca = mockery.mock(ComponentAdapterThatCanManageLifecycle.class);
108        mockery.checking(new Expectations() {{
109                one(ca).start(with(any(PicoContainer.class)));
110                one(ca).stop(with(any(PicoContainer.class)));
111                one(ca).dispose(with(any(PicoContainer.class)));
112                one(ca).start(with(any(Touchable.class)));
113                one(ca).stop(with(any(Touchable.class)));
114                one(ca).dispose(with(any(Touchable.class)));
115        }});
116        return ca;
117    }
118
119    public static interface ComponentAdapterThatCanManageLifecycle extends ComponentAdapter, ChangedBehavior, LifecycleStrategy {
120    }
121
122    static class FooAbstractChangedBehavior extends AbstractBehavior.AbstractChangedBehavior {
123
124        public FooAbstractChangedBehavior(final ComponentAdapter delegate) {
125            super(delegate);
126        }
127
128        public String getDescriptor() {
129            return null;
130        }
131    }
132}
