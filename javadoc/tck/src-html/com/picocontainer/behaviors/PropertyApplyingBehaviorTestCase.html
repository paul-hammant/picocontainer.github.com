010package com.picocontainer.behaviors;
011
012import static org.junit.Assert.assertEquals;
013import static org.junit.Assert.assertNotNull;
014import static org.junit.Assert.assertTrue;
015
016import org.junit.Test;
017import com.picocontainer.testmodel.CoupleBean;
018
019import com.picocontainer.behaviors.PropertyApplying;
020
021/**
022 *
023 * @author greg
024 */
025public class PropertyApplyingBehaviorTestCase {
026    @Test public void testBeanPropertyComponentAdapterCanUsePropertyEditors() {
027        Object c = PropertyApplying.PropertyApplicator.convert(CoupleBean.class.getName(), "a's name:Camilla;b's name:Charles;", this.getClass().getClassLoader());
028        assertNotNull(c);
029        assertTrue(c instanceof CoupleBean);
030        assertEquals("Camilla", ((CoupleBean) c).getPersonA().getName());
031        assertEquals("Charles", ((CoupleBean) c).getPersonB().getName());
032    }
033
034}
