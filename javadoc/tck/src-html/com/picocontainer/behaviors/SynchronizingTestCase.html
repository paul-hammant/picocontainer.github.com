010package com.picocontainer.behaviors;
011
012
013import static com.picocontainer.Characteristics.NO_SYNCHRONIZE;
014import static com.picocontainer.Characteristics.SYNCHRONIZE;
015import static org.junit.Assert.assertNotNull;
016import static org.junit.Assert.assertNull;
017
018import org.junit.Test;
019import com.picocontainer.tck.AbstractComponentFactoryTest;
020
021import com.picocontainer.ComponentFactory;
022import com.picocontainer.MutablePicoContainer;
023import com.picocontainer.PicoBuilder;
024import com.picocontainer.behaviors.Synchronizing;
025import com.picocontainer.injectors.AdaptingInjection;
026
027public class SynchronizingTestCase extends AbstractComponentFactoryTest {
028
029        private final ComponentFactory synchronizing = new Synchronizing().wrap(new AdaptingInjection());
030
031
032        @Override
033        protected ComponentFactory createComponentFactory() {
034                return synchronizing;
035        }
036
037        @Test
038        public void testPicoContainerPropertiesIntegration() {
039                MutablePicoContainer mpc = new PicoBuilder().withBehaviors(new Synchronizing()).build();
040                mpc.as(SYNCHRONIZE).addComponent("a", "This is a test");
041                mpc.as(NO_SYNCHRONIZE).addComponent("b","This is a test");
042
043                assertNotNull(mpc.getComponentAdapter("a").findAdapterOfType(Synchronizing.Synchronized.class));
044                assertNull(mpc.getComponentAdapter("b").findAdapterOfType(Synchronizing.Synchronized.class));
045        }
046
047
048
049}
